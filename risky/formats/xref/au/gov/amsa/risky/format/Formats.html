12  import rx.functions.Action1;

14  import rx.functions.Func1;
15  import au.gov.amsa.util.Files;

17  import com.github.davidmoten.rx.Functions;
18  import com.github.davidmoten.rx.operators.OperatorUnsubscribeEagerly;
19  import com.github.davidmoten.util.Preconditions;
20  import com.google.common.annotations.VisibleForTesting;

22  public final class Formats {
23  
24  	private static final Logger log = LoggerFactory.getLogger(Formats.class);
25  
26  	public static Observable<Integer> transform(final File input,
27  			final File output, Pattern pattern,
28  			final Transformer<Fix, Fix> transformer,
29  			final Action2<List<Fix>, File> fixesWriter,
30  			final Func1<String, String> renamer) {
31  		Preconditions.checkNotNull(input);
32  		Preconditions.checkNotNull(output);
33  		Preconditions.checkNotNull(pattern);
34  		Preconditions.checkNotNull(transformer);
35  
36  		return Observable
37  		// get the files matching the pattern from the directory
38  				.from(Files.find(input, pattern))
39  				// replace the file with a transformed version
40  				.flatMap(new Func1<File, Observable<Integer>>() {
41  
42  					@Override
43  					public Observable<Integer> call(final File file) {
44  						final File outputFile = rebase(file, input, output);
45  						outputFile.getParentFile().mkdirs();
46  						return BinaryFixes.from(file)
47  								// ensure file is closed in case we want to
48  								// rewrite downstream
49  								.lift(OperatorUnsubscribeEagerly
50  										.<Fix> instance())
51  								// to list
52  								.toList()
53  								// flatten
54  								.flatMapIterable(
55  										Functions.<List<Fix>> identity())
56  								// transform the fixes
57  								.compose(transformer)
58  								// make into a list again
59  								.toList()
60  								// replace the file with sorted fixes
61  								.doOnNext(new Action1<List<Fix>>() {
62  									@Override
63  									public void call(List<Fix> list) {
64  										File f = new File(outputFile
65  												.getParentFile(), renamer
66  												.call(outputFile.getName()));
67  										fixesWriter.call(list, f);
68  										log.info("transformed " + file + " to "
69  												+ f);
70  									}
71  								})
72  								// count the fixes
73  								.count();
74  					}
75  				});
76  
77  	}
78  
79  	@VisibleForTesting
80  	static File rebase(File file, File existingParent, File newParent) {
81  		if (file.getAbsolutePath().equals(existingParent.getAbsolutePath()))
82  			return newParent;
83  		else
84  			return new File(rebase(file.getParentFile(), existingParent,
85  					newParent), file.getName());
86  	}
87  
88  }
