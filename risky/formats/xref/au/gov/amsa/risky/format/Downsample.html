1   package au.gov.amsa.risky.format;

3   import java.io.File;
4   import java.util.List;
5   import java.util.concurrent.TimeUnit;
6   import java.util.regex.Pattern;
7   
8   import rx.Observable;
9   import rx.Observable.Transformer;
10  import rx.functions.Action2;
11  import rx.functions.Func2;
12  
13  import com.github.davidmoten.rx.Functions;
14  
15  /**
16   * Assumes input stream is in time order.
17   */
18  public class Downsample implements Transformer<Fix, Fix> {
19  
20  	private long maxTimeBetweenFixesMs;
21  
22  	public Downsample(long minTimeBetweenFixesMs) {
23  		this.maxTimeBetweenFixesMs = minTimeBetweenFixesMs;
24  	}
25  
26  	public static Downsample minTimeStep(long duration, TimeUnit unit) {
27  		return new Downsample(unit.toMillis(duration));
28  	}
29  
30  	@Override
31  	public Observable<Fix> call(Observable<Fix> fixes) {
32  		return fixes.scan(new Func2<Fix, Fix, Fix>() {
33  			@Override
34  			public Fix call(Fix latest, Fix fix) {
35  				if (fix.getTime() < latest.getTime())
36  					throw new RuntimeException("not in ascending time order!");
37  				else if (fix.getMmsi() != latest.getMmsi())
38  					throw new RuntimeException(
39  							"can only downsample a single vessel");
40  				else if (fix.getTime() - latest.getTime() >= maxTimeBetweenFixesMs)
41  					return fix;
42  				else
43  					return latest;
44  			}
45  		})
46  		// throw away repeats
47  				.distinct();
48  	}
49  
50  	public static Observable<Integer> downsample(final File input,
51  			final File output, Pattern pattern, final long duration,
52  			final TimeUnit unit) {
53  		return Formats.transform(input, output, pattern,
54  				Downsample.minTimeStep(duration, unit), FIXES_WRITER,
55  				Functions.<String> identity());
56  	}
57  
58  	private static Action2<List<Fix>, File> FIXES_WRITER = new Action2<List<Fix>, File>() {
59  		@Override
60  		public void call(List<Fix> list, File file) {
61  			BinaryFixesWriter.writeFixes(list, file, false, false);
62  		}
63  	};
64  
65  }
