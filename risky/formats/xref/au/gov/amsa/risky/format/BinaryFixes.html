10  import rx.Observable;
11  import rx.functions.Func1;

13  public final class BinaryFixes {
14  
15  	public static final int BINARY_FIX_BYTES = 31;
16  	public static final short SOG_ABSENT = 1023;
17  	public static final short COG_ABSENT = 3600;
18  	public static final short HEADING_ABSENT = 360;
19  	public static final byte NAV_STATUS_ABSENT = Byte.MAX_VALUE;
20  	public static final int LATENCY_ABSENT = -1;
21  	public static final short SOURCE_ABSENT = 0;
22  	public static final byte ROT_ABSENT = Byte.MIN_VALUE;
23  	public static final byte SOURCE_PRESENT_BUT_UNKNOWN = 1;
24  	protected static final char COMMA = ',';
25  	protected static final byte RATE_OF_TURN_ABSENT = -128;
26  
27  	public static Observable<Fix> from(File file) {
28  		return Observable.create(new BinaryFixesOnSubscribe(file));
29  	}
30  
31  	public static Observable<String> csv(Observable<Fix> fixes) {
32  		return fixes.map(new Func1<Fix, String>() {
33  
34  			@Override
35  			public String call(Fix f) {
36  				StringBuilder s = new StringBuilder();
37  				s.append(f.getLat());
38  				s.append(COMMA);
39  				s.append(f.getLon());
40  				s.append(COMMA);
41  				s.append(new DateTime(f.getTime()).toString());
42  				s.append(COMMA);
43  				s.append(f.getSource().or(SOURCE_ABSENT));
44  				s.append(COMMA);
45  				s.append(f.getLatencySeconds().or(LATENCY_ABSENT));
46  				s.append(COMMA);
47  				s.append(f.getNavigationalStatus().or(
48  						NavigationalStatus.values()[NAV_STATUS_ABSENT]));
49  				s.append(COMMA);
50  				s.append(f.getRateOfTurn().or(RATE_OF_TURN_ABSENT));
51  				s.append(COMMA);
52  				// TODO add the rest of the fields
53  				return s.toString();
54  			}
55  		});
56  	}
57  
58  	public static void write(Fix fix, OutputStream os) {
59  		byte[] bytes = new byte[BINARY_FIX_BYTES];
60  		ByteBuffer bb = ByteBuffer.wrap(bytes);
61  		write(fix, bb);
62  		try {
63  			os.write(bytes);
64  		} catch (IOException e) {
65  			throw new RuntimeException(e);
66  		}
67  	}
68  
69  	public static ByteBuffer createFixByteBuffer() {
70  		return ByteBuffer.allocate(BINARY_FIX_BYTES);
71  	}
72  
73  	public static void write(Fix fix, ByteBuffer bb) {
74  		bb.putFloat(fix.getLat());
75  		bb.putFloat(fix.getLon());
76  		bb.putLong(fix.getTime());
77  		if (fix.getLatencySeconds().isPresent())
78  			bb.putInt(fix.getLatencySeconds().get());
79  		else
80  			bb.putInt(LATENCY_ABSENT);
81  		if (fix.getSource().isPresent())
82  			bb.putShort(fix.getSource().get());
83  		else
84  			bb.putShort(SOURCE_ABSENT);
85  
86  		if (fix.getNavigationalStatus().isPresent())
87  			bb.put((byte) fix.getNavigationalStatus().get().ordinal());
88  		else
89  			bb.put(NAV_STATUS_ABSENT);
90  
91  		// rot
92  		bb.put(ROT_ABSENT);
93  
94  		if (fix.getSpeedOverGroundKnots().isPresent())
95  			bb.putShort((short) Math.round(10 * fix.getSpeedOverGroundKnots()
96  					.get()));
97  		else
98  			bb.putShort(SOG_ABSENT);
99  
100 		if (fix.getCourseOverGroundDegrees().isPresent())
101 			bb.putShort((short) Math.round(10 * fix
102 					.getCourseOverGroundDegrees().get()));
103 		else
104 			bb.putShort(COG_ABSENT);
105 
106 		if (fix.getHeadingDegrees().isPresent())
107 			bb.putShort((short) Math.round(fix.getHeadingDegrees().get()));
108 		else
109 			bb.putShort(HEADING_ABSENT);
110 		if (fix.getAisClass() == AisClass.A)
111 			bb.put((byte) 0);
112 		else
113 			bb.put((byte) 1);
114 	}
115 
116 }
