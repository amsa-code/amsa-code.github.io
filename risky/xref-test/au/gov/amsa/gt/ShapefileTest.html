75      @Test
76      public void testCreatePolygonShapefile() throws IOException {
77          try (InputStreamReader r = new InputStreamReader(
78                  Shapefile.class.getResourceAsStream("/southwest.txt"))) {
79              // read coordinates from classpath
80              List<Coordinate> coords = Strings.lines(r).map(line -> line.trim())
81                      .filter(line -> line.length() > 0).filter(line -> line.charAt(0) != '#')
82                      .doOnNext(System.out::println).map(line -> line.split(" "))
83                      .map(items -> new Coordinate(parseDouble(items[0]), parseDouble(items[1])))
84                      .toList().map(list -> {
85                          List<Coordinate> list2 = Lists.newArrayList(list);
86                          list2.add(new Coordinate(list.get(0).x, list.get(0).y));
87                          return list2;
88                      }).toBlocking().single();
89              ShapefileCreator.createPolygon(coords, new File("target/southwest.shp"));
90          }
91      }
