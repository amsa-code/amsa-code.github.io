3   import java.io.IOException;
4   import java.util.concurrent.CountDownLatch;
5   
6   import org.slf4j.Logger;
7   import org.slf4j.LoggerFactory;

9   import rx.Observable;
10  import rx.Observer;
11  import rx.functions.Func1;
12  import au.gov.amsa.ais.rx.Streams;
13  import au.gov.amsa.navigation.CollisionCandidate;
14  import au.gov.amsa.navigation.CollisionDetector;
15  import au.gov.amsa.navigation.VesselClass;
16  import au.gov.amsa.navigation.VesselPosition;
17  
18  import com.github.davidmoten.rx.slf4j.Logging;
19  
20  public class CollisionDetectorMain {
21  
22  	private static final Logger log = LoggerFactory
23  			.getLogger(CollisionDetectorMain.class);
24  
25  	public static void main(String[] args) throws IOException,
26  			InterruptedException {
27  		CollisionDetector c = new CollisionDetector();
28  		String filename = "/media/analysis/nmea/2013/NMEA_ITU_20130108.gz";
29  		Observable<String> nmea = Streams.nmeaFromGzip(filename)
30  		// do some logging
31  				.lift(Logging.<String> logger().showCount().showMemory()
32  						.every(100000).log());
33  		Observable<VesselPosition> aisPositions = AisVesselPositions.positions(
34  				nmea).filter(onlyClassA);
35  
36  		CountDownLatch latch = new CountDownLatch(1);
37  
38  		c.getCandidates(aisPositions)
39  				// filter
40  				.filter(candidatesMovingWithAtLeastSpeedMetresPerSecond(5 * 0.5144444))
41  				// log
42  				.lift(Logging.<CollisionCandidate> logger().showCount()
43  						.every(1).showValue().showMemory().log())
44  				// subscribe
45  				.subscribe(createSubscriber(latch));
46  		latch.await();
47  	}
48  
49  	private static Func1<CollisionCandidate, Boolean> candidatesMovingWithAtLeastSpeedMetresPerSecond(
50  			final double minSpeedMetresPerSecond) {
51  		return new Func1<CollisionCandidate, Boolean>() {
52  			@Override
53  			public Boolean call(CollisionCandidate c) {
54  				return c.position1().speedMetresPerSecond().get() >= minSpeedMetresPerSecond
55  						&& c.position2().speedMetresPerSecond().get() >= minSpeedMetresPerSecond;
56  			}
57  		};
58  	}
59  
60  	private static Func1<VesselPosition, Boolean> onlyClassA = new Func1<VesselPosition, Boolean>() {
61  
62  		@Override
63  		public Boolean call(VesselPosition p) {
64  			return p.cls() == VesselClass.A;
65  		}
66  	};
67  
68  	private static Observer<CollisionCandidate> createSubscriber(
69  			final CountDownLatch latch) {
70  		return new Observer<CollisionCandidate>() {
71  
72  			@Override
73  			public void onCompleted() {
74  				latch.countDown();
75  			}
76  
77  			@Override
78  			public void onError(Throwable e) {
79  				log.error(e.getMessage(), e);
80  			}
81  
82  			@Override
83  			public void onNext(CollisionCandidate c) {
84  			}
85  		};
86  	}
