21  public final class NmeaGzToBinaryFixesWithMmsiGzMain {
22  
23      private static final Logger log = Logger.getLogger(NmeaGzToBinaryFixesWithMmsiGzMain.class);
24  
25      public static void main(String[] args) throws Exception {
26          // input is one year of nmea lines split into daily files
27          //
28          File inputDir = new File("/home/dxm/AIS");
29          File outputDir = new File(inputDir, "tracks");
30          outputDir.mkdir();
31          Observable<File> inputFiles = Observable.from(inputDir.listFiles()) //
32                  .filter(f -> f.getName().endsWith(".txt.gz")).toSortedList() //
33                  .flatMap(o -> Observable.from(o));
34          AtomicInteger n = new AtomicInteger();
35          inputFiles //
36                  .flatMap(x -> Observable.just(x) //
37                          .doOnNext(f -> convert(f, outputDir, n)) //
38                          .subscribeOn(Schedulers.computation())) //
39                  .toBlocking().subscribe();
40      }
41  
42      private static void convert(File file, File outputDir, AtomicInteger n) {
43          try {
44              log.info("converting " + file.getName() + ", gzipped file size = " + (double) file.length() / 1024 / 1024
45                      + "MB");
46              long t = System.currentTimeMillis();
47              Observable<String> nmea = Streams.nmeaFromGzip(file).onErrorResumeNext(e -> Observable.empty());
48              long count;
49              String outFilename = file.getName().substring(0, file.getName().indexOf(".txt.gz")) + ".track.gz";
50              File outFile = new File(file.getParentFile(), outFilename);
51              try (OutputStream out = new GZIPOutputStream(new BufferedOutputStream(new FileOutputStream(outFile)))) {
52                  count = nmea //
53                          .compose(o -> Streams.extractFixes(o)) //
54                          .filter(x -> inRegion(x)) //
55                          .doOnNext(fix -> BinaryFixes.write(fix, out, BinaryFixesFormat.WITH_MMSI)) //
56                          .countLong() //
57                          .toBlocking() //
58                          .first();
59              }
60              double seconds = (System.currentTimeMillis() - t) / 1000.0;
61              n.incrementAndGet();
62              log.info(n.get() + ": file=" + file.getName() //
63                      + ", count=" + count // "
64                      + ", time=" + seconds + "s" //
65                      + ", rate = " + (count / seconds) + " lines/s" //
66                      + ", outputFileSize = " + outFile.length() / 1024.0 / 1024 + "MB");
67          } catch (IOException e) {
68              throw new UncheckedIOException(e);
69          }
70      }
71  
72      private static boolean inRegion(Fix f) {
73          return f.lat() > -45 && f.lat() < -8 && f.lon() > 108 && f.lon() < 160;
74      }
