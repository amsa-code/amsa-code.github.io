26  public final class BinaryFixesWithMmsiGzCombinedSortMain {
27      
28      private static final Logger log = Logger.getLogger(BinaryFixesWithMmsiGzCombinedSortMain.class);
29  
30      public static void main(String[] args) throws InterruptedException {
31          log.info("starting");
32          File tracks = new File("/home/dxm/AIS/tracks");
33          File combinedSortedTracks = new File("/home/dxm/combinedSortedTracks");
34          combinedSortedTracks.mkdir();
35  
36          
37          AtomicReference<FileFixes> previous = new AtomicReference<>();
38          AtomicInteger n = new AtomicInteger();
39          Observable //
40                  .from(tracks.listFiles()) //
41                  .filter(x -> x.getName().endsWith(".track.gz")) //
42                  .sorted((x,y) -> x.getName().compareTo(y.getName())) //
43                  .doOnNext(x -> log.info("reading " + x)) //
44                  .map(x -> new FileFixes(x, BinaryFixes.from(x, true, BinaryFixesFormat.WITH_MMSI) //
45                          .toList() //
46                          .toBlocking() //
47                          .first()))
48                  .doOnNext(ff -> {
49                      log.info("  extracting previous day fixes from " + ff.file.getName() + ", startTime=" + new Date(ff.startTime));
50                      FileFixes prev = previous.get();
51                      if (prev != null) {
52                          long removed = 0;
53                          long added = 0;
54                          List<Fix> removeThese = new ArrayList<>(32*1024); 
55                          List<Fix> addThese = new ArrayList<>(32*1024);
56                          for (Fix fix: ff.fixes) {
57                              if (fix.time() < ff.startTime) {
58                                  removeThese.add(fix);
59                                  removed++;
60                                  if (fix.time() >= prev.startTime) {
61                                      addThese.add(fix);
62                                      added++;
63                                  }
64                              }
65                          }
66                          TreeSet<Fix> set = new TreeSet<Fix>((x, y) ->  {
67                              if (x.time() == y.time()) {
68                                  return Integer.compare(x.mmsi(), y.mmsi());
69                              } else {
70                                  return Long.compare(x.time(), y.time());
71                              }
72                          });
73                          log.info("  building tree set");
74                          set.addAll(ff.fixes);
75                          log.info("  removing " + removeThese.size() + " from " + ff.file.getName());
76                          set.removeAll(removeThese);
77                          log.info("  copying fixes from set to list");
78                          ff.fixes = new ArrayList<>(set);
79                          log.info("  adding " + addThese.size() + " to " + prev.file.getName());
80                          log.info("  sorting " + prev.file.getName());
81                          prev.fixes.sort((x, y) -> Long.compare(x.time(), y.time()));
82                          log.info("  writing gz for " + prev.file.getName());
83                          writeFixes(combinedSortedTracks, n, prev, removed, added);
84                      }
85                      previous.set(ff);
86                  }).subscribe();
87          if (previous.get() != null) {
88              writeFixes(combinedSortedTracks, n, previous.get(), 0, 0);
89          }
90      }
91  
92      private static void writeFixes(File combinedSortedTracks, AtomicInteger n, FileFixes prev, long removed,
93              long added) {
94          File f = new File(combinedSortedTracks, prev.file.getName());
95          try (OutputStream out = new GZIPOutputStream(new FileOutputStream(f))) {
96              for (Fix fix : prev.fixes) {
97                  BinaryFixes.write(fix, out, BinaryFixesFormat.WITH_MMSI);
98              }
99          } catch (IOException e) {
100             throw new UncheckedIOException(e);
101         }
102         log.info(n.incrementAndGet() + ": removed from next day=" + removed + ", added to this day=" + added + " to " + f);
103     }
104 
105     private static final class FileFixes {
106         final File file;
107         List<Fix> fixes;
108         final long startTime;
109 
110         FileFixes(File file, List<Fix> fixes) {
111             this.file = file;
112             this.fixes = fixes;
113             String datePart = file.getName().substring(0, 10);
114             SimpleDateFormat sdf = new SimpleDateFormat("yyyy-mm-dd");
115             sdf.setTimeZone(TimeZone.getTimeZone("UTC"));
116             try {
117                 startTime = sdf.parse(datePart).getTime();
118             } catch (ParseException e) {
119                 throw new RuntimeException(e);
120             }
121         }
122     }
