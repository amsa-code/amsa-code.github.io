31          Observable
32                  .from(files)
33                  .buffer(Math.max(1, files.size() / Runtime.getRuntime().availableProcessors()) - 1)
34                  .flatMap(
35                          list -> Observable.from(list)
36                                  .lift(Logging.<File> logger().showValue().showMemory().log())
37                                  .concatMap(file -> Streams.extract(Streams.nmeaFromGzip(file))
38                                  //
39                                  // .lift(Logging
40                                  // .<TimestampedAndLine<AisMessage>>
41                                  // logger()
42                                  // .showCount().every(1000000).log())
43                                  //
44                                          .flatMap(aisShipStaticOnly)
45                                          //

47                                          //
48                                          .filter(m -> m instanceof AisShipStaticA)
49                                          //
50                                          .cast(AisShipStaticA.class)
51                                          //
52                                          .distinct(m -> m.getMmsi()))
53                                  //
54                                  .distinct(m -> m.getMmsi())
55                                  //
56                                  .subscribeOn(Schedulers.computation()))
57                  //
58                  .distinct(m -> m.getMmsi())
59                  //
60                  .doOnNext(
61                          m -> {
62                              out.format("%s,%s,%s,%s,%s,%s,%s,%s,%s\n", m.getMmsi(),
63                                      m.getImo().or(-1), "A", m.getShipType(), (float) m
64                                              .getMaximumPresentStaticDraughtMetres(), m
65                                              .getDimensionA().or(-1), m.getDimensionB().or(-1), m
66                                              .getDimensionC().or(-1), m.getDimensionD().or(-1));
67                              out.flush();
68                          })
69                  // go
70                  .count().toBlocking().single();
71          out.close();
72      }

74      private static Func1<TimestampedAndLine<AisMessage>, Observable<TimestampedAndLine<AisShipStatic>>> aisShipStaticOnly = m -> {
75          Optional<Timestamped<AisMessage>> message = m.getMessage();
76          if (message.isPresent() && message.get().message() instanceof AisShipStatic) {
77              @SuppressWarnings("unchecked")
78              TimestampedAndLine<AisShipStatic> m2 = (TimestampedAndLine<AisShipStatic>) (TimestampedAndLine<?>) m;
79              return Observable.just(m2);
80          } else
81              return Observable.empty();
82      };
