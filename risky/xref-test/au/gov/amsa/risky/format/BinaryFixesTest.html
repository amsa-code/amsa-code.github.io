27  	@Test
28  	public void testWriteAndReadBinaryFixes() throws IOException {
29  		File trace = new File("target/123456789.track");
30  		int numFixes = 10000;
31  		writeTrace(trace, numFixes);
32  
33  		System.out.println("wrote " + numFixes + " fixes");
34  
35  		List<Fix> fixes = BinaryFixes.from(trace).toList().toBlocking()
36  				.single();
37  		assertEquals(numFixes, fixes.size());
38  		Fix f = fixes.get(fixes.size() - 1);
39  		assertEquals(123456789, f.getMmsi());
40  		assertEquals(-10.0, f.getLat(), PRECISION);
41  		assertEquals(135, f.getLon(), PRECISION);
42  		assertEquals(1000, f.getTime(), PRECISION);
43  		assertEquals(12, (int) f.getLatencySeconds().get());
44  		assertEquals(1, (int) f.getSource().get());
45  		assertEquals(NavigationalStatus.ENGAGED_IN_FISHING, f
46  				.getNavigationalStatus().get());
47  		assertEquals(7.5, f.getSpeedOverGroundKnots().get(), PRECISION);
48  		assertEquals(45, f.getCourseOverGroundDegrees().get(), PRECISION);
49  		assertEquals(46, f.getHeadingDegrees().get(), PRECISION);
50  		assertEquals(AisClass.B, f.getAisClass());
51  	}

53  	private void writeTrace(File trace, int repetitions) throws IOException {
54  		OutputStream os = new BufferedOutputStream(new FileOutputStream(trace));
55  		Fix fix = new Fix(213456789, -10f, 135f, 1000, of(12), of((short) 1),
56  				of(NavigationalStatus.ENGAGED_IN_FISHING), of(7.5f), of(45f),
57  				of(46f), AisClass.B);
58  		byte[] bytes = new byte[BinaryFixes.BINARY_FIX_BYTES];
59  		ByteBuffer bb = ByteBuffer.wrap(bytes);
60  		BinaryFixes.write(fix, bb);
61  		for (int i = 0; i < repetitions; i++)
62  			os.write(bytes);
63  		os.close();
64  	}

85  	@Test
86  	public void testConcurrencyDemo() {
87  		// using concurrency, count all the fixes across all files in the target
88  		// directory
89  		Observable<File> files = Observable.from(Files.find(new File("target"),
90  				Pattern.compile("\\d+\\.track")));
91  		int count = files
92  		// group the files against each processor
93  				.buffer(Runtime.getRuntime().availableProcessors() - 1)
94  				// do the work per buffer on a separate scheduler
95  				.flatMap(new Func1<List<File>, Observable<Integer>>() {
96  					@Override
97  					public Observable<Integer> call(List<File> list) {
98  						return Observable.from(list)
99  						// count the fixes in each file
100 								.flatMap(countFixes())
101 								// perform concurrently
102 								.subscribeOn(Schedulers.computation());
103 					}
104 				})
105 				// total all the counts
106 				.reduce(0, new Func2<Integer, Integer, Integer>() {
107 					@Override
108 					public Integer call(Integer a, Integer b) {
109 						return a + b;
110 					}
111 				})
112 				// block and get the result
113 				.toBlocking().single();
114 		System.out.println("total fixes = " + count);
115 	}

117 	private Func1<File, Observable<Integer>> countFixes() {
118 		return new Func1<File, Observable<Integer>>() {
119 
120 			@Override
121 			public Observable<Integer> call(File file) {
122 				return BinaryFixes.from(file).count();
123 			}
124 		};
