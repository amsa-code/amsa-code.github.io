18  	public static void main(String[] args) {
19  		// perform a speed test for loading BinaryFixes from disk
20  
21  		Fix.validate = false;
22  		final ConcurrentHashMap<Long, List<Fix>> map = new ConcurrentHashMap<Long, List<Fix>>();
23  		// -downsample-5-mins
24  		List<File> files = Files.find(
25  		        new File("/media/an/binary-fixes/2014-year-downsample-5-mins"),
26  		        Pattern.compile(".*\\.track"));
27  		long t = System.currentTimeMillis();
28  		long count = Observable
29  		// list files
30  		        .from(files)
31  		        // share the load between processors
32  		        .buffer(Math.max(1, files.size() / Runtime.getRuntime().availableProcessors()))
33  		        // count each file asynchronously
34  		        .flatMap(new Func1<List<File>, Observable<Long>>() {
35  			        @Override
36  			        public Observable<Long> call(List<File> list) {
37  				        return Observable.from(list).concatMap(new Func1<File, Observable<Long>>() {
38  					        @Override
39  					        public Observable<Long> call(File file) {
40  						        return BinaryFixes.from(file).countLong();
41  					        }
42  				        })
43  				        // schedule
44  				                .subscribeOn(Schedulers.computation());
45  			        }
46  		        })
47  		        // total counts
48  		        .scan(0L, new Func2<Long, Long, Long>() {
49  			        @Override
50  			        public Long call(Long a, Long b) {
51  				        return a + b;
52  			        }
53  		        })
54  		        // log count so far
55  		        .lift(Logging.<Long> logger().showCount().prefix("records=").showMemory()
56  		                .every(1000).log())
57  		        // get last count (total)
58  		        .last()
59  		        // block and get
60  		        .toBlocking().single();
61  		long elapsed = System.currentTimeMillis() - t;
62  		System.out.println("Map size = " + map.size());
63  		System.out.println("Total records = " + count + ", numPerSecond=" + count * 1000.0
64  		        / elapsed + ", timeMs=" + elapsed);
65  	}
