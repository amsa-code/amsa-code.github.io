181         assertEquals("\\c:1234567,a:1234567890*1F\\" + line,

186     @Test
187     public void testSupplementWithTimeInsertsIntoExistingTagBlock() {
188         String line = "\\s:rEV02,d:1334337321*5A\\!AIVDM,1,1,,B,33:JeT0OjtVls<;fDlbl5CFH2000,0*71";
189         assertEquals(
190                 "\\s:rEV02,d:1334337321,c:1234567,a:1234567890*69\\!AIVDM,1,1,,B,33:JeT0OjtVls<;fDlbl5CFH2000,0*71",
191                 NmeaUtil.supplementWithTime(line, 1234567890));
192         assertEquals("69", NmeaUtil.getChecksum("s:rEV02,d:1334337321,c:1234567,a:1234567890"));
193     }
