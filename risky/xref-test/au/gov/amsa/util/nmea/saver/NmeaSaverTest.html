28  			}
29  		};
30  
31  		File file = new File("target/1970-01-15.txt");
32  		file.delete();
33  		new NmeaSaver(Observable.just(line), new FileFactoryPerDay(new File(
34  				"target")), clock).start(Schedulers.immediate());
35  		Collection<String> lines = FileUtils.readAllLines(file);
36  		assertEquals(1, lines.size());
37  		assertEquals(amendedLine, lines.iterator().next());
38  	}
