3   import static org.junit.Assert.assertEquals;
4   import static org.junit.Assert.assertFalse;
5   
6   import org.junit.Test;
7   
8   public class GridTest {
9   
10  	private static final Options.Builder builder = Options.builder()
11  			.originLat(0).originLon(0).cellSizeDegrees(0.1)
12  			.bounds(new Bounds(0, 100, -60, 175))
13  			.filterBounds(new Bounds(0, 100, -60, 175));
14  	private static final Options options = builder.build();
15  	private static final double PRECISION = 0.0000000001;
16  
17  	@Test
18  	public void testCellAtIfLatAboveBoundsThrowsException() {
19  		Grid grid = new Grid(options);
20  		assertFalse(grid.cellAt(1, 1).isPresent());
21  	}
22  
23  	@Test
24  	public void testCellAtIfLatBelowBoundsThrowsException() {
25  		Grid grid = new Grid(options);
26  		assertFalse(grid.cellAt(-61.0, 1).isPresent());
27  	}
28  
29  	@Test
30  	public void testCellAtIfLonLeftOfBoundsReturnsAbsent() {
31  		Grid grid = new Grid(options);
32  		assertFalse(grid.cellAt(-1, -59).isPresent());
33  	}
34  
35  	@Test
36  	public void testCellAtIfLonRightOfBoundsThrowsException() {
37  		Grid grid = new Grid(options);
38  		assertFalse(grid.cellAt(-1, 176).isPresent());
39  	}
40  
41  	@Test
42  	public void testStartLatGivenOriginLatEqualsTopLeftLatOfBoundsReturnsOriginLat() {
43  		assertEquals(0, Grid.getStartLat(builder.originLat(0.0).build())
44  				.doubleValue(), PRECISION);
45  	}
46  
47  	@Test
48  	public void testStartLatGivenOriginLatAboveTopLeftLatOfBoundsReturnsOriginLat() {
49  		assertEquals(1.0, Grid.getStartLat(builder.originLat(1.0).build())
50  				.doubleValue(), PRECISION);
51  	}
52  
53  	@Test
54  	public void testStartLatGivenOriginLatBelowTopLeftLatOfBoundsReturnsOriginLatMinusEnoughToGetAboveBounds() {
55  		assertEquals(0.1, Grid.getStartLat(builder.originLat(-1.0).build())
56  				.doubleValue(), 0.000000000000000001);
57  	}
58  
59  	@Test
60  	public void testStartLonGivenOriginLonEqualsTopLeftLonOfBoundsReturnsOriginLon() {
61  		assertEquals(100, Grid.getStartLon(builder.originLon(100).build())
62  				.doubleValue(), PRECISION);
63  	}
64  
65  	@Test
66  	public void testStartLonGivenOriginLonLeftOfTopLeftLonOfBoundsReturnsOriginLon() {
67  		assertEquals(99.0, Grid.getStartLon(builder.originLon(99.0).build())
68  				.doubleValue(), PRECISION);
69  	}
70  
71  	@Test
72  	public void testStartLonGivenOriginLonRightOfTopLeftLonOfBoundsReturnsOriginLonMinusEnoughToGetLeftOfBounds() {
73  		assertEquals(99.9, Grid.getStartLon(builder.originLon(101.0).build())
74  				.doubleValue(), 0.000000000000000001);
75  	}
76  
77  	@Test
78  	public void testLeftEdgeLongitude() {
79  		Grid grid = new Grid(options);
80  		Cell cell = grid.cellAt(0, 100).get();
81  		System.out.println(cell);
82  		assertEquals(100.0, cell.leftEdgeLongitude(options), PRECISION);
83  	}
84  
85  	@Test
86  	public void testGridGoingOver180Longitude() {
87  		Options o = options.buildFrom().bounds(new Bounds(0, 100, -60, 175))
88  				.filterBounds(new Bounds(0, 100, -60, -174)).build();
89  		Grid grid = new Grid(o);
90  		System.out.println(grid.rightEdgeLongitude(Cell.cellAt(-10, 100, o)
91  				.get()));
92  		System.out.println(grid.rightEdgeLongitude(Cell.cellAt(-10, 100, o)
93  				.get()));
94  	}
