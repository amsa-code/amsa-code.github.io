42  	@Test
43  	public final void testPredict() {
44  		Position p = new Position(53, 3);
45  		Position p2 = p.predict(100, 30);
46  		Assert.assertEquals(53.77644258276322, p2.getLat(), 0.01);
47  		Assert.assertEquals(3.7609191005595877, p2.getLon(), 0.01);
48  
49  		// large distance tests around the equator
50  
51  		double meanCircumferenceKm = 40041.47;
52  
53  		// start on equator, just East of Greenwich
54  		p = new Position(0, 3);
55  
56  		// quarter circumference, Still in Eastern hemisphere
57  
58  		p2 = p.predict(meanCircumferenceKm / 4.0, 90);
59  		Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
60  		Assert.assertEquals(93.0, p2.getLon(), 0.1);
61  
62  		// half circumference, Now in Western hemisphere,
63  		// just East of the International date line
64  
65  		p2 = p.predict(meanCircumferenceKm / 2.0, 90);
66  		Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
67  		Assert.assertEquals(-177.0, p2.getLon(), 0.1);
68  
69  		// three quarters circumference, Now in Western hemisphere,
70  
71  		p2 = p.predict(3.0 * meanCircumferenceKm / 4.0, 90);
72  		Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
73  		Assert.assertEquals(-87.0, p2.getLon(), 0.1);
74  
75  		// full circumference, back to start
76  		// relax Longitude tolerance slightly
77  
78  		p2 = p.predict(meanCircumferenceKm, 90);
79  		Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
80  		Assert.assertEquals(p.getLon(), p2.getLon(), 0.2);
81  
82  		// same thing but backwards (heading west)
83  
84  		// quarter circumference, Still in western hemisphere
85  
86  		p2 = p.predict(meanCircumferenceKm / 4.0, 270);
87  		Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
88  		Assert.assertEquals(-87.0, p2.getLon(), 0.1);
89  
90  		// half circumference, Now in Western hemisphere,
91  		// just East of the International date line
92  
93  		p2 = p.predict(meanCircumferenceKm / 2.0, 270);
94  		Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
95  		Assert.assertEquals(-177.0, p2.getLon(), 0.1);
96  
97  		// three quarters circumference, Now in eastern hemisphere,
98  
99  		p2 = p.predict(3.0 * meanCircumferenceKm / 4.0, 270);
100 		Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
101 		Assert.assertEquals(93.0, p2.getLon(), 0.1);
102 
103 		// full circumference, back to start
104 		// relax Longitude tolerance slightly
105 
106 		p2 = p.predict(meanCircumferenceKm, 90);
107 		Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
108 		Assert.assertEquals(p.getLon(), p2.getLon(), 0.2);
109 
110 		// OVER THE South POLE
111 		// ===================
112 
113 		// quarter circumference, should be at south pole
114 
115 		p2 = p.predict(meanCircumferenceKm / 4.0, 180);
116 		Assert.assertEquals(-90.0, p2.getLat(), 0.1);
117 
118 		// this next assertion is by no means confident
119 		// expecting 3 but getting it's reciprocal.
120 		// Strange things happen at the pole!!
121 
122 		Assert.assertEquals(-177, p2.getLon(), 0.00001);
123 
124 		// half circumference, should be at the equator
125 		// but in the Western hemisphere
126 
127 		p2 = p.predict(meanCircumferenceKm / 2.0, 180);
128 		Assert.assertEquals(p.getLat(), p2.getLat(), 0.1);
129 		Assert.assertEquals(-177, p2.getLon(), 0.00001);
130 
131 		// 3/4 circumference, should be at the north Pole
132 		// but in the Western hemisphere
133 
134 		p2 = p.predict(3.0 * meanCircumferenceKm / 4.0, 180);
135 		Assert.assertEquals(90.0, p2.getLat(), 0.1);
136 		Assert.assertEquals(p.getLon(), p2.getLon(), 0.00001);
137 
138 		// full circumference, back to start
139 		// relax latitude tolerance slightly
140 
141 		p2 = p.predict(meanCircumferenceKm, 270);
142 		Assert.assertEquals(p.getLat(), p2.getLat(), 0.1);
143 		Assert.assertEquals(p.getLon(), p2.getLon(), 0.2);
144

250 
251 		// Create new position objects
252 		Position p1 = new Position(36.12, -86.67);
253 		Position p2 = new Position(33.94, -118.4);
254 
255 		{
256 			double distanceKm = p1.getDistanceToKm(p2);
257 			double bearingDegrees = p1.getBearingDegrees(p2);
258 			Position p3 = p1.predict(distanceKm * 0.7, bearingDegrees);
259 
260 			// Expected position
261 			Position actual = p1.getPositionAlongPath(p2, 0.7);
262 			// Test expected Lat return position
263 			Assert.assertEquals(p3.getLat(), actual.getLat(), 0.01);
264 			// Test expected Lon return position
265 			Assert.assertEquals(p3.getLon(), actual.getLon(), 0.01);
266 			// Test expected Lat return position
267 			Assert.assertEquals(35.47, actual.getLat(), 0.01);
268 			// Test expected Lon return position
269 			Assert.assertEquals(-109.11, actual.getLon(), 0.01);
270 
271 		}
272 
273 		{
274 			// If start point equals end point then a proportion
275 			// along the path should equal the start point.
276 			Position actual = p1.getPositionAlongPath(p1, 0.7);
277 			// Test expected Lat return position
278 			Assert.assertEquals(p1.getLat(), actual.getLat(), 0.01);
279 			// Test expected Lon return position
280 			Assert.assertEquals(p1.getLon(), actual.getLon(), 0.01);
281 		}
282

284 			// If proportion is 0.0 then should return start point
285 			Position actual = p1.getPositionAlongPath(p2, 0.0);
286 			// Test expected Lat return position
287 			Assert.assertEquals(p1.getLat(), actual.getLat(), 0.01);
288 			// Test expected Lon return position
289 			Assert.assertEquals(p1.getLon(), actual.getLon(), 0.01);
290 		}
291 
292 		{
293 			// If proportion is 1.0 then should return end point
