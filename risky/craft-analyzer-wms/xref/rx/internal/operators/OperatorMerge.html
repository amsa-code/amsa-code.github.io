35  /**
36   * Flattens a list of {@link Observable}s into one {@code Observable}, without any transformation.
37   * <p>
38   * <img width="640" height="380" src="https://raw.githubusercontent.com/wiki/ReactiveX/RxJava/images/rx-operators/merge.png" alt="">
39   * <p>
40   * You can combine the items emitted by multiple {@code Observable}s so that they act like a single {@code Observable}, by using the merge operation.
41   * 
42   * @param <T>
43   *            the type of the items emitted by both the source and merged {@code Observable}s
44   */
45  public class OperatorMerge<T> implements Operator<T, Observable<? extends T>> {
46  
47      /*
48       * benjchristensen => This class is complex and I'm not a fan of it despite writing it. I want to give some background
49       * as to why for anyone who wants to try and help improve it.
50       * 
51       * One of my first implementations that added backpressure support (Producer.request) was fairly elegant and used a simple
52       * queue draining approach. It was simple to understand as all onNext were added to their queues, then a single winner
53       * would drain the queues, similar to observeOn. It killed the Netflix API when I canaried it. There were two problems:
54       * (1) performance and (2) object allocation overhead causing massive GC pressure. Remember that merge is one of the most
55       * used operators (mostly due to flatmap) and is therefore critical to and a limiter of performance in any application.
56       * 
57       * All subsequent work on this class and the various fast-paths and branches within it have been to achieve the needed functionality
58       * while reducing or eliminating object allocation and keeping performance acceptable.
59       * 
60       * This has meant adopting strategies such as:
61       * 
62       * - ring buffers instead of growable queues
63       * - object pooling
64       * - skipping request logic when downstream does not need backpressure
65       * - ScalarValueQueue for optimizing synchronous single-value Observables
66       * - adopting data structures that use Unsafe (and gating them based on environment so non-Oracle JVMs still work)
67       * 
68       * It has definitely increased the complexity and maintenance cost of this class, but the performance gains have been significant.
69       * 
70       * The biggest cost of the increased complexity is concurrency bugs and reasoning through what's going on.
71       * 
72       * I'd love to have contributions that improve this class, but keep in mind the performance and GC pressure.
73       * The benchmarks I use are in the JMH OperatorMergePerf class. GC memory pressure is tested using Java Flight Recorder
74       * to track object allocation.
75       */

93      private static final class MergeSubscriber<T> extends Subscriber<Observable<? extends T>> {
94          final NotificationLite<T> on = NotificationLite.instance();
95          final Subscriber<? super T> actual;
96          private final MergeProducer<T> mergeProducer;
97          private int wip;
98          private boolean completed;
99          private final boolean delayErrors;
100         private ConcurrentLinkedQueue<Throwable> exceptions;
101 
102         private volatile SubscriptionIndexedRingBuffer<InnerSubscriber<T>> childrenSubscribers;
103 
104         private RxRingBuffer scalarValueQueue = null;
105 
106         /* protected by lock on MergeSubscriber instance */
107         private int missedEmitting = 0;
108         private boolean emitLock = false;
109 
110         /**
111          * Using synchronized(this) for `emitLock` instead of ReentrantLock or AtomicInteger is faster when there is no contention.
112          * 
113          * <pre> {@code
114          * Using ReentrantLock:
115          * r.o.OperatorMergePerf.merge1SyncStreamOfN           1000  thrpt         5    44185.294     1295.565    ops/s
116          * 
117          * Using synchronized(this):
118          * r.o.OperatorMergePerf.merge1SyncStreamOfN           1000  thrpt         5    79715.981     3704.486    ops/s
119          * 
120          * Still slower though than allowing concurrency:
121          * r.o.OperatorMergePerf.merge1SyncStreamOfN           1000  thrpt         5   149331.046     4851.290    ops/s
122          * } </pre>
123          */

125         public MergeSubscriber(Subscriber<? super T> actual, boolean delayErrors) {
126             super(actual);
127             this.actual = actual;
128             this.mergeProducer = new MergeProducer<T>(this);
129             this.delayErrors = delayErrors;
130             // decoupled the subscription chain because we need to decouple and control backpressure
131             actual.add(this);
132             actual.setProducer(mergeProducer);
133         }
134 
135         @Override
136         public void onStart() {
137             // we request backpressure so we can handle long-running Observables that are enqueueing, such as flatMap use cases
138             // we decouple the Producer chain while keeping the Subscription chain together (perf benefit) via super(actual)
139             request(RxRingBuffer.SIZE);
140         }

142         /*
143          * This is expected to be executed sequentially as per the Rx contract or it will not work.
144          */
145         @Override
146         public void onNext(Observable<? extends T> t) {
147             if (t instanceof ScalarSynchronousObservable) {
148                 ScalarSynchronousObservable<? extends T> t2 = (ScalarSynchronousObservable<? extends T>)t;
149                 handleScalarSynchronousObservable(t2);
150             } else {
151                 if (t == null || isUnsubscribed()) {
152                     return;
153                 }
154                 synchronized (this) {
155                     // synchronized here because `wip` can be concurrently changed by children Observables
156                     wip++;
157                 }
158                 handleNewSource(t);
159             }
160         }
161 
162         private void handleNewSource(Observable<? extends T> t) {
163             if (childrenSubscribers == null) {
164                 // lazily create this only if we receive Observables we need to subscribe to
165                 childrenSubscribers = new SubscriptionIndexedRingBuffer<InnerSubscriber<T>>();
166                 add(childrenSubscribers);
167             }
168             MergeProducer<T> producerIfNeeded = null;
169             // if we have received a request then we need to respect it, otherwise we fast-path
170             if (mergeProducer.requested != Long.MAX_VALUE) {
171                 /**
172                  * <pre> {@code
173                  * With this optimization:
174                  * 
175                  * r.o.OperatorMergePerf.merge1SyncStreamOfN      1000  thrpt         5    57100.080     4686.331    ops/s
176                  * r.o.OperatorMergePerf.merge1SyncStreamOfN   1000000  thrpt         5       60.875        1.622    ops/s
177                  *  
178                  * Without this optimization:
179                  * 
180                  * r.o.OperatorMergePerf.merge1SyncStreamOfN      1000  thrpt         5    29863.945     1858.002    ops/s
181                  * r.o.OperatorMergePerf.merge1SyncStreamOfN   1000000  thrpt         5       30.516        1.087    ops/s
182                  * } </pre>
183                  */
184                 producerIfNeeded = mergeProducer;
185             }
186             InnerSubscriber<T> i = new InnerSubscriber<T>(this, producerIfNeeded);
187             i.sindex = childrenSubscribers.add(i);
188             t.unsafeSubscribe(i);
189             if (!isUnsubscribed()) {
190                 request(1);
191             }
192         }

194         private void handleScalarSynchronousObservable(ScalarSynchronousObservable<? extends T> t) {
195             // fast-path for scalar, synchronous values such as Observable.from(int)
196             /**
197              * Without this optimization:
198              * 
199              * <pre> {@code
200              * Benchmark                                          (size)   Mode   Samples        Score  Score error    Units
201              * r.o.OperatorMergePerf.oneStreamOfNthatMergesIn1         1  thrpt         5  2,418,452.409   130572.665    ops/s
202              * r.o.OperatorMergePerf.oneStreamOfNthatMergesIn1      1000  thrpt         5     5,690.456       94.958    ops/s
203              * r.o.OperatorMergePerf.merge1SyncStreamOfN         1000000  thrpt         5          takes too long
204              * 
205              * With this optimization:
206              * 
207              * r.o.OperatorMergePerf.merge1SyncStreamOfN               1  thrpt         5  5,475,300.198   156741.334    ops/s
208              * r.o.OperatorMergePerf.merge1SyncStreamOfN            1000  thrpt         5    68,932.278     1311.023    ops/s
209              * r.o.OperatorMergePerf.merge1SyncStreamOfN         1000000  thrpt         5       64.405        0.611    ops/s
210              * } </pre>
211              * 
212              */
213             if (mergeProducer.requested == Long.MAX_VALUE) {
214                 handleScalarSynchronousObservableWithoutRequestLimits(t);
215             } else {
216                 handleScalarSynchronousObservableWithRequestLimits(t);
217             }
218         }
219 
220         private void handleScalarSynchronousObservableWithoutRequestLimits(ScalarSynchronousObservable<? extends T> t) {
221             T value = t.get();
222             if (getEmitLock()) {
223                 boolean moreToDrain;
224                 try {
225                     actual.onNext(value);
226                 } finally {
227                     moreToDrain = releaseEmitLock();
228                 }
229                 if (moreToDrain) {
230                     drainQueuesIfNeeded();
231                 }
232                 request(1);
233                 return;
234             } else {
235                 initScalarValueQueueIfNeeded();
236                 try {
237                     scalarValueQueue.onNext(value);
238                 } catch (MissingBackpressureException e) {
239                     onError(e);
240                 }
241                 return;
242             }
243         }
244 
245         private void handleScalarSynchronousObservableWithRequestLimits(ScalarSynchronousObservable<? extends T> t) {
246             if (getEmitLock()) {
247                 boolean emitted = false;
248                 boolean moreToDrain;
249                 boolean isReturn = false;
250                 try {
251                     long r = mergeProducer.requested;
252                     if (r > 0) {
253                         emitted = true;
254                         actual.onNext(t.get());
255                         MergeProducer.REQUESTED.decrementAndGet(mergeProducer);
256                         // we handle this Observable without ever incrementing the wip or touching other machinery so just return here
257                         isReturn = true;
258                     }
259                 } finally {
260                     moreToDrain = releaseEmitLock();
261                 }
262                 if (moreToDrain) {
263                     drainQueuesIfNeeded();
264                 }
265                 if (emitted) {
266                     request(1);
267                 }
268                 if (isReturn) {
269                     return;
270                 }
271             }
272 
273             // if we didn't return above we need to enqueue
274             // enqueue the values for later delivery
275             initScalarValueQueueIfNeeded();
276             try {
277                 scalarValueQueue.onNext(t.get());
278             } catch (MissingBackpressureException e) {
279                 onError(e);
280             }
281         }
282 
283         private void initScalarValueQueueIfNeeded() {
284             if (scalarValueQueue == null) {
285                 scalarValueQueue = RxRingBuffer.getSpmcInstance();
286                 add(scalarValueQueue);
287             }
288         }
289 
290         private synchronized boolean releaseEmitLock() {
291             emitLock = false;
292             if (missedEmitting == 0) {
293                 return false;
294             } else {
295                 return true;
296             }
297         }
298 
299         private synchronized boolean getEmitLock() {
300             if (emitLock) {
301                 missedEmitting++;
302                 return false;
303             } else {
304                 emitLock = true;
305                 missedEmitting = 0;
306                 return true;
307             }
308         }
309 
310         private boolean drainQueuesIfNeeded() {
311             while (true) {
312                 if (getEmitLock()) {
313                     int emitted = 0;
314                     boolean moreToDrain;
315                     try {
316                         emitted = drainScalarValueQueue();
317                         drainChildrenQueues();
318                     } finally {
319                         moreToDrain = releaseEmitLock();
320                     }
321                     // request outside of lock
322                     if (emitted > 0) {
323                         request(emitted);
324                     }
325                     if (!moreToDrain) {
326                         return true;
327                     }
328                     // otherwise we'll loop and get whatever was added
329                 } else {
330                     return false;
331                 }
332             }
333         }
334 
335         int lastDrainedIndex = 0;
336 
337         /**
338          * ONLY call when holding the EmitLock.
339          */
340         private void drainChildrenQueues() {
341             if (childrenSubscribers != null) {
342                 lastDrainedIndex = childrenSubscribers.forEach(DRAIN_ACTION, lastDrainedIndex);
343             }
344         }
345 
346         /**
347          * ONLY call when holding the EmitLock.
348          */
349         private int drainScalarValueQueue() {
350             if (scalarValueQueue != null) {
351                 long r = mergeProducer.requested;
352                 int emittedWhileDraining = 0;
353                 if (r < 0) {
354                     // drain it all
355                     Object o = null;
356                     while ((o = scalarValueQueue.poll()) != null) {
357                         on.accept(actual, o);
358                         emittedWhileDraining++;
359                     }
360                 } else if (r > 0) {
361                     // drain what was requested
362                     long toEmit = r;
363                     for (int i = 0; i < toEmit; i++) {
364                         Object o = scalarValueQueue.poll();
365                         if (o == null) {
366                             break;
367                         } else {
368                             on.accept(actual, o);
369                             emittedWhileDraining++;
370                         }
371                     }
372                     // decrement the number we emitted from outstanding requests
373                     MergeProducer.REQUESTED.getAndAdd(mergeProducer, -emittedWhileDraining);
374                 }
375                 return emittedWhileDraining;
376             }
377             return 0;
378         }

380         final Func1<InnerSubscriber<T>, Boolean> DRAIN_ACTION = new Func1<InnerSubscriber<T>, Boolean>() {
381 
382             @Override
383             public Boolean call(InnerSubscriber<T> s) {
384                 if (s.q != null) {
385                     long r = mergeProducer.requested;
386                     int emitted = s.drainQueue();
387                     if (emitted > 0) {
388                         s.requestMore(emitted);
389                     }
390                     if (emitted == r) {
391                         // we emitted as many as were requested so stop the forEach loop
392                         return Boolean.FALSE;
393                     }
394                 }
395                 return Boolean.TRUE;
396             }
397 
398         };
399 
400         @Override
401         public void onError(Throwable e) {
402             if (!completed) {
403                 completed = true;
404                 innerError(e, true);
405             }
406         }
407         
408         private void innerError(Throwable e, boolean parent) {
409             if (delayErrors) {
410                 synchronized (this) {
411                     if (exceptions == null) {
412                         exceptions = new ConcurrentLinkedQueue<Throwable>();
413                     }
414                 }
415                 exceptions.add(e);
416                 boolean sendOnComplete = false;
417                 synchronized (this) {
418                     if (!parent) {
419                         wip--;
420                     }
421                     if ((wip == 0 && completed) || (wip < 0)) {
422                         sendOnComplete = true;
423                     }
424                 }
425                 if (sendOnComplete) {
426                     drainAndComplete();
427                 }
428             } else {
429                 actual.onError(e);
430             }
431         }
432 
433         @Override
434         public void onCompleted() {
435             boolean c = false;
436             synchronized (this) {
437                 completed = true;
438                 if (wip == 0 && (scalarValueQueue == null || scalarValueQueue.isEmpty())) {
439                     c = true;
440                 }
441             }
442             if (c) {
443                 // complete outside of lock
444                 drainAndComplete();
445             }
446         }
447 
448         void completeInner(InnerSubscriber<T> s) {
449             boolean sendOnComplete = false;
450             synchronized (this) {
451                 wip--;
452                 if (wip == 0 && completed) {
453                     sendOnComplete = true;
454                 }
455             }
456             childrenSubscribers.remove(s.sindex);
457             if (sendOnComplete) {
458                 drainAndComplete();
459             }
460         }
461 
462         private void drainAndComplete() {
463             drainQueuesIfNeeded(); // TODO need to confirm whether this is needed or not
464             if (delayErrors) {
465                 Queue<Throwable> es = null;
466                 synchronized (this) {
467                     es = exceptions;
468                 }
469                 if (es != null) {
470                     if (es.isEmpty()) {
471                         actual.onCompleted();
472                     } else if (es.size() == 1) {
473                         actual.onError(es.poll());
474                     } else {
475                         actual.onError(new CompositeException(es));
476                     }
477                 } else {
478                     actual.onCompleted();
479                 }
480             } else {
481                 actual.onCompleted();
482             }
483         }

487     private static final class MergeProducer<T> implements Producer {
488 
489         private final MergeSubscriber<T> ms;
490 
491         public MergeProducer(MergeSubscriber<T> ms) {
492             this.ms = ms;
493         }
494 
495         private volatile long requested = 0;
496         @SuppressWarnings("rawtypes")
497         static final AtomicLongFieldUpdater<MergeProducer> REQUESTED = AtomicLongFieldUpdater.newUpdater(MergeProducer.class, "requested");
498 
499         @Override
500         public void request(long n) {
501             if (requested == Long.MAX_VALUE) {
502                 return;
503             }
504             if (n == Long.MAX_VALUE) {
505                 requested = Long.MAX_VALUE;
506             } else {
507                 REQUESTED.getAndAdd(this, n);
508                 if (ms.drainQueuesIfNeeded()) {
509                     boolean sendComplete = false;
510                     synchronized (ms) {
511                         if (ms.wip == 0 && ms.scalarValueQueue != null && ms.scalarValueQueue.isEmpty()) {
512                             sendComplete = true;
513                         }
514                     }
515                     if (sendComplete) {
516                         ms.drainAndComplete();
517                     }
518                 }
519             }
520         }

524     private static final class InnerSubscriber<T> extends Subscriber<T> {
525         public int sindex;
526         final MergeSubscriber<T> parentSubscriber;
527         final MergeProducer<T> producer;
528         /** Make sure the inner termination events are delivered only once. */
529         @SuppressWarnings("unused")
530         volatile int terminated;
531         @SuppressWarnings("rawtypes")
532         static final AtomicIntegerFieldUpdater<InnerSubscriber> ONCE_TERMINATED = AtomicIntegerFieldUpdater.newUpdater(InnerSubscriber.class, "terminated");
533 
534         private final RxRingBuffer q = RxRingBuffer.getSpmcInstance();
535 
536         public InnerSubscriber(MergeSubscriber<T> parent, MergeProducer<T> producer) {
537             this.parentSubscriber = parent;
538             this.producer = producer;
539             add(q);
540             request(q.capacity());
541         }
542 
543         @Override
544         public void onNext(T t) {
545             emit(t, false);
546         }
547 
548         @Override
549         public void onError(Throwable e) {
550             // it doesn't go through queues, it immediately onErrors and tears everything down
551             if (ONCE_TERMINATED.compareAndSet(this, 0, 1)) {
552                 parentSubscriber.innerError(e, false);
553             }
554         }
555 
556         @Override
557         public void onCompleted() {
558             if (ONCE_TERMINATED.compareAndSet(this, 0, 1)) {
559                 emit(null, true);
560             }
561         }
562 
563         public void requestMore(long n) {
564             request(n);
565         }
566 
567         private void emit(T t, boolean complete) {
568             boolean drain = false;
569             boolean enqueue = true;
570             /**
571              * This optimization to skip the queue is messy ... but it makes a big difference in performance when merging a single stream
572              * with many values, or many intermittent streams without contention. It doesn't make much of a difference if there is contention.
573              * 
574              * Below are some of the relevant benchmarks to show the difference.
575              * 
576              * <pre> {@code
577              * With this fast-path:
578              * 
579              * Benchmark                                          (size)   Mode   Samples        Score  Score error    Units
580              * r.o.OperatorMergePerf.merge1SyncStreamOfN               1  thrpt         5  5344143.680   393484.592    ops/s
581              * r.o.OperatorMergePerf.merge1SyncStreamOfN            1000  thrpt         5    83582.662     4293.755    ops/s +++
582              * r.o.OperatorMergePerf.merge1SyncStreamOfN         1000000  thrpt         5       73.889        4.477    ops/s +++
583              * 
584              * r.o.OperatorMergePerf.mergeNSyncStreamsOfN              1  thrpt         5  5799265.333   199205.296    ops/s +
585              * r.o.OperatorMergePerf.mergeNSyncStreamsOfN           1000  thrpt         5       62.655        2.521    ops/s +++
586              * 
587              * r.o.OperatorMergePerf.mergeTwoAsyncStreamsOfN           1  thrpt         5    76925.616     4909.174    ops/s
588              * r.o.OperatorMergePerf.mergeTwoAsyncStreamsOfN        1000  thrpt         5     3634.977      242.469    ops/s
589              * 
590              * Without:
591              * 
592              * Benchmark                                          (size)   Mode   Samples        Score  Score error    Units
593              * r.o.OperatorMergePerf.merge1SyncStreamOfN               1  thrpt         5  5099295.678   159539.842    ops/s
594              * r.o.OperatorMergePerf.merge1SyncStreamOfN            1000  thrpt         5    18196.671    10053.298    ops/s
595              * r.o.OperatorMergePerf.merge1SyncStreamOfN         1000000  thrpt         5       19.184        1.028    ops/s
596              * 
597              * r.o.OperatorMergePerf.mergeNSyncStreamsOfN              1  thrpt         5  5591612.719   591821.763    ops/s
598              * r.o.OperatorMergePerf.mergeNSyncStreamsOfN           1000  thrpt         5       21.018        3.251    ops/s
599              * 
600              * r.o.OperatorMergePerf.mergeTwoAsyncStreamsOfN           1  thrpt         5    72692.073    18395.031    ops/s
601              * r.o.OperatorMergePerf.mergeTwoAsyncStreamsOfN        1000  thrpt         5     4379.093      386.368    ops/s
602              * } </pre>
603              * 
604              * It looks like it may cause a slowdown in highly contended cases (like 'mergeTwoAsyncStreamsOfN' above) as instead of just
605              * putting in the queue, it attempts to get the lock. We are optimizing for the non-contended case.
606              */
607             if (parentSubscriber.getEmitLock()) {
608                 long emitted = 0;
609                 enqueue = false;
610                 try {
611                     // drain the queue if there is anything in it before emitting the current value
612                     emitted += drainQueue();
613                     //                    }
614                     if (producer == null) {
615                         // no backpressure requested
616                         if (complete) {
617                             parentSubscriber.completeInner(this);
618                         } else {
619                             try {
620                                 parentSubscriber.actual.onNext(t);
621                             } catch (Throwable e) {
622                                 // special error handling due to complexity of merge
623                                 onError(OnErrorThrowable.addValueAsLastCause(e, t));
624                             }
625                             emitted++;
626                         }
627                     } else {
628                         // this needs to check q.count() as draining above may not have drained the full queue
629                         // perf tests show this to be okay, though different queue implementations could perform poorly with this
630                         if (producer.requested > 0 && q.count() == 0) {
631                             if (complete) {
632                                 parentSubscriber.completeInner(this);
633                             } else {
634                                 try {
635                                     parentSubscriber.actual.onNext(t);
636                                 } catch (Throwable e) {
637                                     // special error handling due to complexity of merge
638                                     onError(OnErrorThrowable.addValueAsLastCause(e, t));
639                                 }
640                                 emitted++;
641                                 MergeProducer.REQUESTED.decrementAndGet(producer);
642                             }
643                         } else {
644                             // no requests available, so enqueue it
645                             enqueue = true;
646                         }
647                     }
648                 } finally {
649                     drain = parentSubscriber.releaseEmitLock();
650                 }
651                 // request upstream what we just emitted
652                 if(emitted > 0) {
653                     request(emitted);
654                 }
655             }
656             if (enqueue) {
657                 enqueue(t, complete);
658                 drain = true;
659             }
660             if (drain) {
661                 /**
662                  * This extra check for whether to call drain is ugly, but it helps:
663                  * <pre> {@code
664                  * Without:
665                  * r.o.OperatorMergePerf.mergeNSyncStreamsOfN     1000  thrpt         5       61.812        1.455    ops/s
666                  * 
667                  * With:
668                  * r.o.OperatorMergePerf.mergeNSyncStreamsOfN     1000  thrpt         5       78.795        1.766    ops/s
669                  * } </pre>
670                  */
671                 parentSubscriber.drainQueuesIfNeeded();
672             }
673         }
674 
675         private void enqueue(T t, boolean complete) {
676             try {
677                 if (complete) {
678                     q.onCompleted();
679                 } else {
680                     q.onNext(t);
681                 }
682             } catch (MissingBackpressureException e) {
683                 onError(e);
684             }
685         }
686 
687         private int drainRequested() {
688             int emitted = 0;
689             // drain what was requested
690             long toEmit = producer.requested;
691             Object o;
692             for (int i = 0; i < toEmit; i++) {
693                 o = q.poll();
694                 if (o == null) {
695                     // no more items
696                     break;
697                 } else if (q.isCompleted(o)) {
698                     parentSubscriber.completeInner(this);
699                 } else {
700                     try {
701                         if (!q.accept(o, parentSubscriber.actual)) {
702                             emitted++;
703                         }
704                     } catch (Throwable e) {
705                         // special error handling due to complexity of merge
706                         onError(OnErrorThrowable.addValueAsLastCause(e, o));
707                     }
708                 }
709             }
710 
711             // decrement the number we emitted from outstanding requests
712             MergeProducer.REQUESTED.getAndAdd(producer, -emitted);
713             return emitted;
714         }
715 
716         private int drainAll() {
717             int emitted = 0;
718             // drain it all
719             Object o;
720             while ((o = q.poll()) != null) {
721                 if (q.isCompleted(o)) {
722                     parentSubscriber.completeInner(this);
723                 } else {
724                     try {
725                         if (!q.accept(o, parentSubscriber.actual)) {
726                             emitted++;
727                         }
728                     } catch (Throwable e) {
729                         // special error handling due to complexity of merge
730                         onError(OnErrorThrowable.addValueAsLastCause(e, o));
731                     }
732                 }
733             }
734             return emitted;
735         }
736 
737         private int drainQueue() {
738             if (producer != null) {
739                 return drainRequested();
740             } else {
741                 return drainAll();
742             }
743         }
