17  	@Test
18  	public void testReaderFromInputStream() {
19  		IhsReader r = new IhsReader();
20  		Observable<Ship> o = r.from(IhsReaderTest.class
21  				.getResourceAsStream("/ShipData.xml"));
22  		Ship ship = o.last().toBlocking().single();
23  		System.out.println(ship);
24  		assertEquals("4544107", ship.getImo());
25  		assertEquals("235101211", ship.getMmsi().get());
26  		assertEquals("Miscellaneous", ship.getType2().get());
27  		assertEquals("Other Activities", ship.getType3().get());
28  		assertEquals("Patrol Vessel", ship.getType4().get());
29  		assertEquals("Patrol Vessel", ship.getType5().get());
30  		assertEquals(525, (long) ship.getGrossTonnage().get());
31  		assertEquals("RI", ship.getClassificationSocietyCode().get());
32  		assertEquals("GBI", ship.getFlagCode().get());
33  		assertEquals("0860230", ship.getGroupBeneficialOwnerCompanyCode().get());
34  		assertEquals("GBI", ship.getGroupBeneficialOwnerCountryOfDomicileCode()
35  				.get());
36  		assertEquals("FIN", ship.getCountryOfBuildCode().get());
37  		assertEquals(2002, (int) ship.getYearOfBuild().get());
38  		assertEquals(1, (int) ship.getMonthOfBuild().get());
39  		assertEquals(1445.0, ship.getDeadweightTonnage().get(), PRECISION);
40  		assertEquals("B34H2SQ", ship.getStatCode5().get());
41  		assertEquals(49.7, ship.getLengthOverallMetres().get(), PRECISION);
42  		assertEquals(7.5, ship.getBreadthMetres().get(), PRECISION);
43  		assertFalse(ship.getDisplacementTonnage().isPresent());
44  		assertEquals(3.7, ship.getDraughtMetres().get(), PRECISION);
45  		assertEquals(22.0, ship.getSpeedKnots().get(), PRECISION);
46  		assertEquals("2014-01-16T14:47:29.780Z", ship.getLastUpdateTime().get()
47  				.toString());
48  	}
