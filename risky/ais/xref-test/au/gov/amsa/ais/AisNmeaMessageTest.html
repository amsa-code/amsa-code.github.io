17      @Test
18      public void testAisNmeaMessage() {
19          String m = "$BSVDM,2,2,5,B,00000000008,2*33";
20          AisNmeaMessage a = new AisNmeaMessage(m);
21          assertEquals("B", a.getChannel());
22          assertEquals("$BSVDM", a.getFormat());
23          assertEquals(2, a.getFragmentCount());
24          assertEquals(2, a.getFragmentNumber());
25          assertEquals("5", a.getSequentialMessageId());
26          assertEquals("33", a.getChecksum());
27          assertEquals(Talker.UNKNOWN, a.getTalker());
28          assertNotNull(a.getNmea());
29          a.getMessage();
30  
31      }

59      @Test
60      public void testGetTimeObtainsTimeFromTagBlock() {
61          String line = "\\g:1-2-1234,s:r3669961,c:1120959341*51\\$BSVDM,2,2,5,B,00000000008,2*33";
62          AisNmeaMessage m = new AisNmeaMessage(line);
