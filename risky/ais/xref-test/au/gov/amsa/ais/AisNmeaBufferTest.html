32  	private void checkTwoLinesAreBufferedAndAggregateIsReturned(
33  			AisNmeaBuffer buffer) {
34  		String line1 = "\\g:1-2-1130,c:1334278696*29\\!BSVDM,2,1,0,A,55DSBL02<Dm7<I`OP005<T4r0hTiT00000000016=hJ<855f?>kV`54Qh000,0*0B";
35  		String line2 = "\\g:2-2-1130*5E\\!BSVDM,2,2,0,A,00000000002,0*3D";
36  		assertNull(buffer.add(NmeaUtil.parseNmea(line1)));
37  		NmeaMessage m = buffer.add(NmeaUtil.parseNmea(line2));
38  		// check that column 5 in line2 is appended to column 5 in line 1 and
39  		// line count and line number are 1 and tag block is as for line1.
40  		System.out.println(m.toLine());
41  		assertEquals(
42  				"\\g:1-1-1130,c:1334278696*2A\\!BSVDM,1,1,0,A,55DSBL02<Dm7<I`OP005<T4r0hTiT00000000016=hJ<855f?>kV`54Qh00000000000002,0*3A",
43  				m.toLine());
44  		System.out
45  				.println("msg="
46  						+ new AisMessageParser()
47  								.parse("55DSBL02<Dm7<I`OP005<T4r0hTiT00000000016=hJ<855f?>kV`54Qh00000000000002"));
48  	}

53  		String line2 = "\\g:1-2-3987,c:1333239510*20\\!BSVDM,2,1,0,A,54`98002>?A1`<AGD00lEBr0PD5@PE:1<4hiT01CKh`IC4w8NKjCPj1Ck`2k,0*7C";
54  		assertNull(buffer.add(NmeaUtil.parseNmea(line1)));
55  		NmeaMessage m = buffer.add(NmeaUtil.parseNmea(line2));
56  		// check that column 5 in line2 is appended to column 5 in line 1 and
57  		// line count and line number are 1 and tag block is as for line1.
58  		System.out.println(m.toLine());
59  		assertEquals(

61  				m.toLine());
62  	}

72  	@Test
73  	public void checkBufferMaxSizeIsHonoured() {
74  		AisNmeaBuffer buffer = new AisNmeaBuffer(2);
75  		// this message
76  		String line = "\\g:2-2-3985*5A\\!AIVDM,2,2,9,B,PFRC88888888880,2*29";
77  		assertNull(buffer.add(NmeaUtil.parseNmea(line)));
78  		assertEquals(1, buffer.size());
79  		String line2 = "\\g:2-2-1130*5E\\!BSVDM,2,2,0,A,00000000002,0*3D";
80  		assertNull(buffer.add(NmeaUtil.parseNmea(line2)));
81  		assertEquals(2, buffer.size());
82  		String line3 = "\\g:2-2-1132*5E\\!BSVDM,2,2,0,A,00000000002,0*3D";
83  		assertNull(buffer.add(NmeaUtil.parseNmea(line3)));
84  		assertEquals(2, buffer.size());
85  	}
