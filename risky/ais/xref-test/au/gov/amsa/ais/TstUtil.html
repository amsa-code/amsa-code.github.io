19  	static void handleAisStream(InputStream is,
20  			PrintStream messagesWithTimestamp, PrintStream out,
21  			PrintStream console) throws IOException {
22  		SimpleDateFormat sdf = new SimpleDateFormat("HH:mm");
23  		DecimalFormat df = new DecimalFormat("00");
24  		sdf.setTimeZone(TimeZone.getTimeZone("UTC"));
25  		BufferedReader br = new BufferedReader(new InputStreamReader(is));
26  		String line;
27  		while ((line = br.readLine()) != null) {
28  			out.println(line);
29  			messagesWithTimestamp.println(line + ","
30  					+ System.currentTimeMillis());
31  			try {
32  				AisNmeaMessage nmea = new AisNmeaMessage(line);
33  				Timestamped<AisMessage> message = nmea.getTimestampedMessage();
34  				out.println(message);
35  				out.println("talker=" + nmea.getTalker().getDescription());
36  				if (message.message().getMessageId() <= 3 && message.message().getMessageId() > 0) {
37  					Communications comms = ((HasCommunications) message)
38  							.getCommunications();
39  					if (comms.getMinutesUtc() != null) {
40  						String commsTime = df.format(comms.getHourUtc()) + ":"
41  								+ df.format(comms.getMinuteUtc());
42  						String arrivalTime = sdf.format(new Date(message.time()));
43  						String output = "comms,arrival=" + commsTime + ","
44  								+ arrivalTime;
45  						if (!commsTime.equals(arrivalTime)) {
46  							console.println(output + " " + line);
47  						}
48  						out.println(output + "\t" + line);
49  					}
50  				}
51  			} catch (RuntimeException e) {
52  				out.println("-------------- ERROR: " + e.getMessage());
53  			}
54  		}
55  		br.close();
56  	}

62  	static NmeaStreamProcessorListener createLoggingListener(
63  			final PrintStream out) {
64  		return new NmeaStreamProcessorListener() {
65  
66  			int count = 0;
67  
68  			@Override
69  			public void message(String line, long time) {
70  				count++;
71  				out.println(count + " " + line + " -> " + new Date(time));
72  			}
73  
74  			@Override
75  			public void timestampNotFound(String line, Long arrivalTime) {
76  				String commsDate = "";
77  				String className = "";
78  				try {
79  					AisNmeaMessage nmea = new AisNmeaMessage(line);
80  					AisMessage message = nmea.getMessage();
81  					className = message.getClass().getSimpleName();
82  					if (message instanceof HasCommunications) {
83  						Communications comms = ((HasCommunications) message)
84  								.getCommunications();
85  						if (comms.getMinutesUtc() != null) {
86  							DecimalFormat df = new DecimalFormat("00");
87  							commsDate = df.format(comms.getHourUtc()) + ":"
88  									+ df.format(comms.getMinuteUtc());
89  						}
90  					}
91  				} catch (AisParseException e) {
92  
93  				}
94  				out.println(count++ + " " + line + " -> "
95  						+ new Date(arrivalTime) + " NOT FOUND " + className
96  						+ " " + commsDate);
97  			}
98  
99  			@Override
100 			public void invalidNmea(String line, long arrivalTime,
101 					String message) {
102 				out.println("invalid nmea: " + line);
103 			}
104 		};
105 	}

107 	static void process(NmeaReader con, NmeaStreamProcessorListener listener,
108 			final PrintStream out) {
109 
110 		NmeaStreamProcessor p = new NmeaStreamProcessor(listener, true);
111 		for (String line : con.read()) {
112 			// System.out.println(line);
113 			NmeaMessage nmea = NmeaUtil.parseNmea(line);
114 			long time = Long.parseLong(nmea.getItems().get(
115 					nmea.getItems().size() - 1));
116 			int i = line.lastIndexOf(",");
117 			p.line(line.substring(0, i), time);
118 		}
119 		out.println("buffer remaining=" + p.getBuffer().size());
120 		out.println("buffer=");
121 		for (LineAndTime s : p.getBuffer())
122 			out.println(s);
123 
124 	}
