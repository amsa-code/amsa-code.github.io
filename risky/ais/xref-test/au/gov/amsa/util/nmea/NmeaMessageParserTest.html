15  	@Test
16  	public void testNmeaMessageParser() {
17  		String line = "\\g:3-3-1234*hh\\$ABVSI,r3669961,1,013536.96326433,1386,-98,,*hh";
18  		NmeaMessage n = NmeaUtil.parseNmea(line);
19  		assertEquals("3-3-1234", n.getSentenceGroupingFromTagBlock());
20  		assertNull(n.getSource());
21  		assertNull(n.getDestination());
22  		assertNull(n.getLineCount());
23  		assertNull(n.getRelativeTimeMillis());
24  		assertNull(n.getText());
25  		assertNull(n.getUnixTimeMillis());
26  		assertEquals("$ABVSI", n.getItems().get(0));
27  		assertEquals("r3669961", n.getItems().get(1));
28  		assertEquals(8, n.getItems().size());
29  	}

32  	public void testParsingOfTimeAndSourceInTagBlock() {
33  		String line = "\\g:1-2-1234,s:r3669961,c:1120959341*hh\\!ABVDM,1,1,1,B,…..,0*hh";
34  		NmeaMessage n = NmeaUtil.parseNmea(line);
35  		assertEquals("1-2-1234", n.getSentenceGroupingFromTagBlock());
36  		assertEquals(1, (int) n.getSentenceNumber());
37  		assertEquals(2, (int) n.getSentenceCount());
38  		assertEquals("1234", n.getSentenceGroupId());
39  		assertEquals("r3669961", n.getSource());
40  		assertNull(n.getDestination());
41  		assertNull(n.getLineCount());
42  		assertNull(n.getRelativeTimeMillis());
43  		assertNull(n.getText());
44  		assertEquals(1120959341000L, (long) n.getUnixTimeMillis());
45  	}

154 	public void testTagBlockThatIncludesAValueWithAColonIsParsedCorrectly() {
155 		String msg = "\\s:rEV06,c:1418371240,i:|X=1|D=1|T=41985.3864759144|P=10.225.253.129:25479|R=IN|*54\\!AIVDM,1,1,,A,13IMtL01BC4mJ7uurMNWP6;>08H:,0*03";
156 		NmeaMessage n = new NmeaMessageParser().parse(msg);
157 		LinkedHashMap<String, String> tags = n.getTags();
158 		assertEquals("rEV06", tags.get("s"));
159 		assertEquals("1418371240", tags.get("c"));
160 		assertEquals(
161 				"|X=1|D=1|T=41985.3864759144|P=10.225.253.129:25479|R=IN|",
162 				tags.get("i"));
163 		assertEquals(3, tags.size());
164 	}
