3   import java.util.HashSet;

6   public final class MmsiValidator2 {
7   
8   	private final Set<Long> badIdentifiers = new HashSet<Long>();
9   
10  	public static MmsiValidator2 INSTANCE = new MmsiValidator2();
11  
12  	private MmsiValidator2() {
13  		badIdentifiers.add(123456789L);
14  		badIdentifiers.add(987654321L);
15  		badIdentifiers.add(111111111L);
16  		badIdentifiers.add(999999999L);
17  		// multiple ships use these mmsi numbers
18  		badIdentifiers.add(107374182L);
19  		badIdentifiers.add(503499100L);
20  		badIdentifiers.add(503000000L);
21  		badIdentifiers.add(777777777L);
22  		badIdentifiers.add(333333333L);
23  		badIdentifiers.add(525123456L);
24  		badIdentifiers.add(273000000L);
25  		badIdentifiers.add(525000000L);
26  		badIdentifiers.add(100000000L);
27  		badIdentifiers.add(1193046L);
28  	}
29  
30  	/**
31  	 * Returns true if and only if the <code>mmsi</code> is a series of 9 digits
32  	 * and is not one of a set of bad identifiers e.g. 123456789.
33  	 * 
34  	 * @param mmsi
35  	 * @return
36  	 */
37  	public boolean isValid(long mmsi) {
38  		return mmsi <= 999999999L && mmsi >= 1000000L && !badIdentifiers.contains(mmsi);
39  	}
40  }
