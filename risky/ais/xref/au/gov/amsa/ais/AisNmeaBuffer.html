1   package au.gov.amsa.ais;

3   import java.util.Collection;
4   import java.util.Collections;
5   import java.util.Comparator;
6   import java.util.LinkedHashMap;
7   import java.util.List;
8   
9   import au.gov.amsa.util.nmea.NmeaMessage;
10  import au.gov.amsa.util.nmea.NmeaMessageParseException;
11  
12  import com.google.common.collect.LinkedHashMultimap;
13  import com.google.common.collect.Lists;
14  
15  public class AisNmeaBuffer {
16  
17  	private static final int AIS_MESSAGE_COL_NO = 5;
18  	private static final int MIN_NUM_COLS_FOR_LINE_TO_BE_AGGREGATED = 6;
19  	private final int columnToAggregate;
20  	private final int maxBufferSize;
21  	private final LinkedHashMultimap<String, NmeaMessage> buffer;
22  
23  	public AisNmeaBuffer(int columnToAggregate, int maxBufferSize) {
24  		this.columnToAggregate = columnToAggregate;
25  		this.maxBufferSize = maxBufferSize;
26  		buffer = LinkedHashMultimap.create();
27  	}
28  
29  	public AisNmeaBuffer(int maxBufferSize) {
30  		this(AIS_MESSAGE_COL_NO, maxBufferSize);
31  	}
32  
33  	/**
34  	 * Returns the complete message only once the whole group of messages has
35  	 * arrived otherwise returns null.
36  	 * 
37  	 * @param nmea
38  	 * @return
39  	 */
40  	public synchronized NmeaMessage add(NmeaMessage nmea) {
41  		List<String> items = nmea.getItems();
42  		if (items.size() < MIN_NUM_COLS_FOR_LINE_TO_BE_AGGREGATED)
43  			return nmea;
44  		if (nmea.isSingleSentence()) {
45  			return nmea;
46  		} else {
47  			String groupId = nmea.getSentenceGroupId();
48  			buffer.put(groupId, nmea);
49  			// trim the oldest if we have reached max buffer size
50  			if (buffer.size() > maxBufferSize)
51  				buffer.removeAll(buffer.keySet().iterator().next());
52  			int numGroupMessages = nmea.getSentenceCount();
53  			int numGroupMessagesSoFar = buffer.get(groupId).size();
54  			if (numGroupMessagesSoFar == numGroupMessages) {
55  				// we have all messages in that group now so concatenate
56  				List<NmeaMessage> list = orderMessages(buffer.get(groupId));
57  				NmeaMessage concatenatedMessage = concatenateMessages(list);
58  				buffer.removeAll(groupId);
59  				return concatenatedMessage;
60  			} else
61  				return null;
62  		}
63  	}
64  
65  	/**
66  	 * Returns the aggregated message or if an {@link NmeaMessageParseException}
67  	 * occurs returns null.
68  	 * 
69  	 * @param list
70  	 * @return
71  	 */
72  	private NmeaMessage concatenateMessages(List<NmeaMessage> list) {
73  		NmeaMessage first = list.get(0);
74  		// concatenate column 5 and use row 1 tag block
75  
76  		// copy cols so we can modify, don't want to affect the original nmea
77  		// message
78  		List<String> cols = Lists.newArrayList(first.getItems());
79  
80  		// copy tags so we can modify
81  		LinkedHashMap<String, String> tags = new LinkedHashMap<String, String>(
82  				first.getTags());
83  
84  		StringBuilder s = new StringBuilder();
85  		for (NmeaMessage t : list) {
86  			s.append(t.getItems().get(columnToAggregate));
87  		}
88  		cols.set(columnToAggregate, s.toString());
89  		// set num sentences to be 1 and current sentence number to be 1
90  		cols.set(1, "1");
91  		cols.set(2, "1");
92  
93  		tags.put("g", "1-1-" + first.getSentenceGroupId());
94  		try {
95  			NmeaMessage message = new NmeaMessage(tags, cols);
96  			return message;
97  		} catch (NmeaMessageParseException e) {
98  			return null;
99  		}
100 	}
101 
102 	private static List<NmeaMessage> orderMessages(Collection<NmeaMessage> c) {
103 		List<NmeaMessage> list = Lists.newArrayList(c);
104 		Collections.sort(list, new Comparator<NmeaMessage>() {
105 			@Override
106 			public int compare(NmeaMessage a, NmeaMessage b) {
107 				return a.getSentenceNumber().compareTo(b.getSentenceNumber());
108 			}
109 		});
110 		return list;
111 	}
112 
113 	public int size() {
114 		return buffer.size();
115 	}
116 }
