4   import au.gov.amsa.ais.message.AisBaseStation;
5   import au.gov.amsa.ais.message.AisMessageOther;
6   import au.gov.amsa.ais.message.AisPositionA;
7   import au.gov.amsa.ais.message.AisPositionB;
8   import au.gov.amsa.ais.message.AisPositionBExtended;
9   import au.gov.amsa.ais.message.AisShipStaticA;

11  /**
12   * Parses AIS messages (as they are taken from the 5th column in the NMEA
13   * message).
14   * 
15   * @author dxm
16   * 
17   */
18  public class AisMessageParser {
19  
20  	private final AisExtractorFactory factory;
21  
22  	/**
23  	 * Constructor.
24  	 */
25  	public AisMessageParser() {
26  		this(Util.getAisExtractorFactory());
27  	}
28  
29  	/**
30  	 * Constructor.
31  	 * 
32  	 * @param factory
33  	 */
34  	public AisMessageParser(AisExtractorFactory factory) {
35  		this.factory = factory;
36  	}
37  
38  	/**
39  	 * Returns an {@link AisMessage} from the string representation of the
40  	 * message as per 1371-4 IMO specification (as per the appropriate column in
41  	 * the NMEA message). Sets source to null.
42  	 * 
43  	 * @param message
44  	 * @return
45  	 */
46  	public AisMessage parse(String message, int padBits) {
47  		return parse(message, null, padBits);
48  	}
49  
50  	/**
51  	 * Returns an {@link AisMessage} from the string representation of the
52  	 * message as per 1371-4 IMO specification (as per the appropriate column in
53  	 * the NMEA message).
54  	 * 
55  	 * @param message
56  	 * @param source
57  	 * @return
58  	 */
59  	public AisMessage parse(String message, String source, int padBits) {
60  		AisExtractor extractor = factory.create(message, 0, padBits);
61  		int id = extractor.getMessageId();
62  		if (Util.isClassAPositionReport(id)) {
63  			return new AisPositionA(message, source, padBits);
64  		} else if (id == 4)
65  			return new AisBaseStation(message, source, padBits);
66  		else if (id == 5)
67  			return new AisShipStaticA(message, source, padBits);
68  		else if (id == 18)
69  			return new AisPositionB(message, source, padBits);
70  		else if (id == 19)
71  			return new AisPositionBExtended(message, source, padBits);
72  		else if (id == 21)
73  			return new AisAidToNavigation(message, source, padBits);
74  		else
75  			return new AisMessageOther(id, source, padBits);
76  	}
77  
78  }
