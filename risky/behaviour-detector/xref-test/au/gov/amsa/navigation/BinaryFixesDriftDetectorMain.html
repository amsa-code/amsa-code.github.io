31      private static final String DRIFT_CANDIDATES_TXT = "target/drift-candidates.txt";
32      private static final Logger log = LoggerFactory.getLogger(BinaryFixesDriftDetectorMain.class);

34      public static void main(String[] args) {
35          new File(DRIFT_CANDIDATES_TXT).delete();
36          VesselPosition.validate = false;
37          FixImpl.validate = false;
38          List<File> files = Files.find(new File("/media/an/binary-fixes/2015"),
39                  Pattern.compile(".*\\.track"));
40          log.info("files=" + files.size());
41          final AtomicLong num = new AtomicLong();
42          int count = Observable
43          // list files
44                  .from(files)
45                  // exclude invalid mmsi
46                  .filter(onlyValidMmsis())
47                  // share the load between processors
48                  .buffer(Math.max(1,
49                          files.size()
50                                  / (Math.max(1, Runtime.getRuntime().availableProcessors() - 1))))
51                  // search each list of files for drift detections
52                  .flatMap(detectDrift(num, Schedulers.computation()))
53                  // count
54                  .reduce(0, BinaryFixesDriftDetectorMain.<Integer> add()).toBlocking().single();
55          log.info("drift detections = " + count);
56      }

58      private static Func1<? super File, Boolean> onlyValidMmsis() {
59          return file -> MmsiValidator2.INSTANCE.isValid(Long.parseLong(file.getName().substring(0,
60                  file.getName().indexOf('.'))));
61      }

63      @SuppressWarnings("unchecked")
64      private static <T extends Number> Func2<T, T, T> add() {
65          return (a, b) -> {
66              if (a instanceof Integer)
67                  return (T) (Number) (a.intValue() + b.intValue());
68              else if (a instanceof Long)
69                  return (T) (Number) (a.longValue() + b.longValue());
70              else if (a instanceof Double)
71                  return (T) (Number) (a.doubleValue() + b.doubleValue());
72              else if (a instanceof Float)
73                  return (T) (Number) (a.floatValue() + b.floatValue());
74              else if (a instanceof Byte)
75                  return (T) (Number) (a.byteValue() + b.byteValue());
76              else if (a instanceof Short)
77                  return (T) (Number) (a.shortValue() + b.shortValue());
78              else
79                  throw new RuntimeException("not implemented");
80          };
81      }

83      private static Func1<List<File>, Observable<Integer>> detectDrift(final AtomicLong num,
84              final Scheduler scheduler) {
85          return list -> {
86              return Observable.from(list)
87                      // files to drift detections
88                      .concatMap(
89                              file -> {
90                                  return BinaryFixes.from(file)
91                                  // log count
92                                          .doOnNext(logCount(num))
93                                          //
94                                          .compose(Fixes.ignoreOutOfOrderFixes(false))
95                                          // detect drift
96                                          .compose(DriftDetector.detectDrift())
97                                          // downsample to min 5 minutes between
98                                          // reports but ensure that start of
99                                          // drift is always included
100                                         .compose(
101                                                 Downsample.<DriftCandidate> minTimeStep(5,
102                                                         TimeUnit.MINUTES, isStartOfDrift()))
103                                         // onNext
104                                         .doOnNext(ON_NEXT)
105                                         // log on error
106                                         .doOnError(e -> {
107                                             log.error(file + ":" + e.getMessage(), e);
108                                         })
109                                         // count
110                                         .count();
111                             })
112                     // schedule
113                     .subscribeOn(scheduler);
114         };
115     }

117     private static Func1<DriftCandidate, Boolean> isStartOfDrift() {
118         return c -> c.driftingSince() == c.fix().time();
119     }
120 
121     private static final Action1<DriftCandidate> ON_NEXT = new Action1<DriftCandidate>() {
122         final char COMMA = ',';
123         final Charset UTF8 = Charset.forName("UTF-8");
124 
125         @Override
126         public synchronized void call(DriftCandidate c) {
127             try {
128                 FileOutputStream os = new FileOutputStream(DRIFT_CANDIDATES_TXT, true);
129                 Fix f = c.fix();
130                 StringBuilder s = new StringBuilder();
131                 s.append(f.mmsi());
132                 s.append(COMMA);
133                 s.append(f.lat());
134                 s.append(COMMA);
135                 s.append(f.lon());
136                 s.append(COMMA);
137                 s.append(f.time());
138                 s.append(COMMA);
139                 s.append(f.aisClass());
140                 s.append(COMMA);
141                 s.append(f.courseOverGroundDegrees().get());
142                 s.append(COMMA);
143                 s.append(f.headingDegrees().get());
144                 s.append(COMMA);
145                 s.append(f.speedOverGroundKnots().get());
146                 s.append(COMMA);
147                 s.append(f.navigationalStatus().isPresent() ? f.navigationalStatus().get()
148                         .toString() : "");
149                 s.append(COMMA);
150                 s.append(c.driftingSince());
151                 s.append('\n');
152                 os.write(s.toString().getBytes(UTF8));
153                 os.close();
154             } catch (IOException e) {
155                 throw new RuntimeException(e);
156             }
157         }
158     };
159 
160     private static Action1<Fix> logCount(final AtomicLong num) {
161         return p -> {
162             long n = num.incrementAndGet();
163             if (n % 1000000 == 0) {
164                 log.info((n / 1000000.0) + "m");
165             }
166         };
167     }
