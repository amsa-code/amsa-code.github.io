34          ConcurrentHashMap<Integer, Long> map = new ConcurrentHashMap<>();
35          int count = Observable.from(files) //
36                  // .doOnNext(System.out::println) //
37                  // .doOnNext(System.out::println)
38                  .concatMap(file -> BinaryFixes.from(file)
39                          .filter(f -> f.courseOverGroundDegrees().isPresent()
40                                  && f.headingDegrees().isPresent())
41                          .map(f -> Math.abs(Position.getBearingDifferenceDegrees(
42                                  f.courseOverGroundDegrees().get(), f.headingDegrees().get()))) //
43                          .doOnNext(x -> {
44                              int bucket = (int) Math.round(x);
45                              map.compute(bucket, (n, num) -> num == null ? 1 : num + 1);
46                          })//
47          ) //
48                  .count().toBlocking().single();
49  
50          System.out.println("count=" + count);
51          for (Integer key : new TreeSet<Integer>(map.keySet())) {
52              System.out.println(key + "\t" + map.get(key));
53          }
54      }
