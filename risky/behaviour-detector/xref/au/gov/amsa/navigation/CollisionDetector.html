53      private static Func1<VesselPosition, Region> toRegion() {
54          return new Func1<VesselPosition, Region>() {
55  
56              @Override
57              public Region call(VesselPosition p) {
58                  double maxLat = 15;
59                  double minLat = -50;
60                  double minLon = -70;
61                  double maxLon = 179;
62                  int numRegions = Runtime.getRuntime().availableProcessors();
63                  int x = (int) Math.floor((p.lon() - minLon) / (maxLon - minLon) * numRegions);
64                  double lonCellSize = (maxLon - minLon) / numRegions;
65                  double longitudeDelta;
66                  if (Math.abs(minLat) > Math.abs(maxLat))
67                      longitudeDelta = longitudeDelta(minLat);
68                  else
69                      longitudeDelta = longitudeDelta(maxLat);
70                  return new Region(maxLat, minLon + x * lonCellSize - longitudeDelta, minLat, minLon
71                          + (x + 1) * lonCellSize + longitudeDelta);
72              }
73          };
74      }

76      public static Observable<CollisionCandidate> getCandidatesForAStream(
77              Observable<VesselPosition> o) {
78          // make a window of recent positions indexed spatially
79          return o.scan(new State(), nextState())
80                  // log
81                  .lift(Logging
82                          .<State> logger()
83                          .showCount("positions")
84                          .showRateSince("rate (pos/s)", TimeUnit.SECONDS.toMillis(10))
85                          .showRateSinceStart("overall rate")
86                          .every(10000)
87                          .showValue()
88                          .value(state -> "state.map.size=" + state.mapSize() + ", state.rtree.size="
89                                  + state.tree().size()).log())
90                  // report collision candidates from each window for the latest
91                  // reported position
92                  .flatMap(toCollisionCandidatesForPosition())
93                  // group by id of first candidate
94                  .groupBy(byIdPair())
95                  // only show if repeated
96                  .flatMap(onlyRepeating());
97      }

99      private static Func2<State, VesselPosition, State> nextState() {
100         return (state, p) -> state.nextState(MAX_TIME_INTERVAL_MS, p);

103     private static Func1<State, Observable<CollisionCandidate>> toCollisionCandidatesForPosition() {
104         return state -> {
105             if (!state.last().isPresent())
106                 return Observable.empty();
107             else {
108                 return toCollisionCandidatesForPosition(state);
109             }
110         };
111     }
112 
113     private static Observable<CollisionCandidate> toCollisionCandidatesForPosition(State state) {
114         final VesselPosition p = state.last().get();
115         final Optional<VesselPosition> next = state.nextPosition();
116 
117         // use the spatial index to get positions physically near the latest
118         // position report
119 
120         // setup a region around the latest position report to search with a
121         // decent delta).
122         double longitudeDelta = longitudeDelta(p.lat());
123         Rectangle searchRegion = rectangle(p.lon() - longitudeDelta, p.lat() - LATITUDE_DELTA,
124                 p.lon() + longitudeDelta, p.lat() + LATITUDE_DELTA);
125 
126         // find nearby vessels within time constraints and cache them
127         Observable<VesselPosition> near = state.tree()
128         // search the R-tree
129                 .search(searchRegion)
130                 // get just the vessel position
131                 .map(toVesselPosition)
132                 // only accept positions with time close to p
133                 .filter(aroundInTime(p, MAX_TIME_INTERVAL_MS));
134 
135         final Observable<TreeSet<VesselPosition>> othersByVessel = near
136         // only those vessels with different id as latest position report
137                 .filter(not(isVessel(p.id())))
138                 // group by individual vessel
139                 .groupBy(byId())
140                 // sort the positions by time
141                 .flatMap(toSortedSet());
142 
143         Observable<CollisionCandidate> collisionCandidates = othersByVessel
144                 .flatMap(toCollisionCandidates2(p, next));
145 
146         return collisionCandidates;
147     }

149     private static <T> Func1<T, Boolean> not(final Func1<T, Boolean> f) {
150         return t -> !f.call(t);

153     private static double longitudeDelta(double lat) {
154         return LATITUDE_DELTA / cos(toRadians(lat));
155     }

157     private static Func1<GroupedObservable<IdentifierPair, CollisionCandidate>, Observable<? extends CollisionCandidate>> onlyRepeating() {
158         return g -> g.buffer(2).flatMap(isSmallTimePeriod());
159     }

161     private static Func1<List<CollisionCandidate>, Observable<CollisionCandidate>> isSmallTimePeriod() {
162         return list -> {
163             Optional<Long> min = Optional.absent();
164             Optional<Long> max = Optional.absent();
165             for (CollisionCandidate c : list) {
166                 if (!min.isPresent() || c.position1().time() < min.get())
167                     min = Optional.of(c.position1().time());
168                 if (!max.isPresent() || c.position1().time() > max.get())
169                     max = Optional.of(c.position1().time());
170             }
171             if (max.get() - min.get() < TimeUnit.MINUTES.toMillis(5))
172                 return from(list);
173             else
174                 return empty();
175         };
176     }

178     private static Func1<? super CollisionCandidate, IdentifierPair> byIdPair() {
179         return c -> new IdentifierPair(c.position1().id(), c.position2().id());
180     }

182     private static Func1<TreeSet<VesselPosition>, Observable<CollisionCandidate>> toCollisionCandidates2(
183             final VesselPosition p, final Optional<VesselPosition> next) {
184         return set -> {
185             Optional<VesselPosition> other = Optional.fromNullable(set.lower(p));
186             if (other.isPresent()) {
187                 Optional<Times> times = p.intersectionTimes(other.get());
188                 if (times.isPresent()) {
189                     Optional<Long> tCollision = plus(times.get().leastPositive(), p.time());
190                     if (tCollision.isPresent()
191                             && tCollision.get() < p.time() + MAX_TIME_INTERVAL_MS) {
192                         Optional<VesselPosition> otherNext = Optional.fromNullable(set.higher(other
193                                 .get()));
194                         if (otherNext.isPresent() && otherNext.get().time() < tCollision.get())
195                             return empty();
196                         else if (next.isPresent() && next.get().time() < tCollision.get())
197                             return empty();
198                         else
199                             return just(new CollisionCandidate(p, other.get(), tCollision.get()));
200                     } else
201                         return empty();
202                 } else
203                     return empty();
204             } else
205                 return empty();
206         };
207     }

209     private static Optional<Long> plus(Optional<Long> a, long b) {
210         if (a.isPresent())
211             return Optional.of(a.get() + b);
212         else
213             return Optional.absent();
214     }

216     private static Func1<GroupedObservable<Identifier, VesselPosition>, Observable<TreeSet<VesselPosition>>> toSortedSet() {
217         return g -> g.toList().map(
218                 singleVesselPositions -> {
219                     TreeSet<VesselPosition> set = new TreeSet<VesselPosition>(
220                             Comparators.timeIdMessageIdComparator);
221                     set.addAll(singleVesselPositions);
222                     return set;
223                 });
224     }

226     private static Func1<VesselPosition, Identifier> byId() {
227         return position -> position.id();

230     private static Func1<VesselPosition, Boolean> aroundInTime(final VesselPosition position,
231             final long maxTimeIntervalMs) {
232         return p -> Math.abs(p.time() - position.time()) <= maxTimeIntervalMs;
233     }

235     private static Func1<VesselPosition, Boolean> isVessel(final Identifier id) {
236         return p -> p.id().equals(id);
