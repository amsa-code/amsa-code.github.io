1   package au.gov.amsa.navigation;

3   import rx.Observable;
4   import rx.Observable.Transformer;
5   import rx.functions.Func1;
6   
7   import com.google.common.base.Preconditions;
8   
9   public class DriftingDetector {
10  
11  	private static final int HEADING_COG_DIFFERENCE_MIN = 70;
12  	private static final int HEADING_COG_DIFFERENCE_MAX = 110;
13  	protected static final double MAX_DRIFTING_SPEED_KNOTS = 4;
14  	protected static final double MIN_DRIFTING_SPEED_KNOTS = 0.3;
15  
16  	public Observable<VesselPosition> getCandidates(Observable<VesselPosition> o) {
17  		return o.filter(isCandidate());
18  	}
19  	
20  	private static class DriftingTransformer implements Transformer<VesselPosition,VesselPosition>{
21  		
22  		private DriftingDetector d = new DriftingDetector();
23  		
24  		@Override
25  		public Observable<VesselPosition> call(Observable<VesselPosition> o) {
26  			return d.getCandidates(o);
27  		}
28  	}
29  	
30  	public static DriftingTransformer detectDrift() {
31  		return new DriftingTransformer();
32  	}
33  	
34  	private Func1<VesselPosition, Boolean> isCandidate() {
35  
36  		return new Func1<VesselPosition, Boolean>() {
37  
38  			@Override
39  			public Boolean call(VesselPosition p) {
40  				if (p.cogDegrees().isPresent()
41  						&& p.headingDegrees().isPresent()) {
42  					double diff = diff(p.cogDegrees().get(), p.headingDegrees()
43  							.get());
44  					return diff >= HEADING_COG_DIFFERENCE_MIN
45  							&& diff <= HEADING_COG_DIFFERENCE_MAX
46  							&& p.speedMetresPerSecond().get() <= MAX_DRIFTING_SPEED_KNOTS * 0.5144444
47  							&& p.speedMetresPerSecond().get() > MIN_DRIFTING_SPEED_KNOTS;
48  				} else
49  					return false;
50  			}
51  		};
52  	}
53  
54  	static double diff(double a, double b) {
55  		Preconditions.checkArgument(a >= 0 && a < 360,
56  				"must be between 0 and 360" + a);
57  		Preconditions.checkArgument(b >= 0 && b < 360,
58  				"must be between 0 and 360: " + b);
59  		double value;
60  		if (a < b)
61  			value = a + 360 - b;
62  		else
63  			value = a - b;
64  		if (value > 180)
65  			return 360 - value;
66  		else
67  			return value;
68  
69  	};
70  
71  }
