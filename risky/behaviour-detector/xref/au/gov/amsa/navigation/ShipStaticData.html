47  	public static Observable<Info> fromAndClose(Reader reader) {
48  		return Observable.using(() -> reader, r -> from(r),
49  				Checked.a1(r -> r.close()));
50  	}
51  
52  	public static Observable<Info> from(Reader reader) {
53  		return Strings
54  				.lines(reader)
55  				// ignore comments
56  				.filter(line -> !line.startsWith("#"))
57  				.map(line -> line.trim())
58  				.filter(line -> line.length() > 0)
59  				.map(line -> line.split(","))
60  				.map(items -> {
61  					int i = 0;
62  					long mmsi = Long.parseLong(items[i++]);
63  					String imoTemp = items[i++];
64  					Optional<String> imo;
65  					if (imoTemp.trim().length() == 0)
66  						imo = Optional.absent();
67  					else
68  						imo = Optional.of(imoTemp);
69  					AisClass cls = items[i++].equals("A") ? AisClass.A
70  							: AisClass.B;
71  					int type = Integer.parseInt(items[i++]);
72  					Optional<Integer> shipType = type == -1 ? absent()
73  							: of(type);
74  					float draft = Float.parseFloat(items[i++]);
75  					Optional<Float> maxDraftMetres = draft == -1 ? absent()
76  							: of(draft);
77  					int a = Integer.parseInt(items[i++]);
78  					int b = Integer.parseInt(items[i++]);
79  					int c = Integer.parseInt(items[i++]);
80  					int d = Integer.parseInt(items[i++]);
81  					Optional<Integer> dimensionAMetres = a == -1 ? absent()
82  							: of(a);
83  					Optional<Integer> dimensionBMetres = b == -1 ? absent()
84  							: of(b);
85  					Optional<Integer> dimensionCMetres = c == -1 ? absent()
86  							: of(c);
87  					Optional<Integer> dimensionDMetres = d == -1 ? absent()
88  							: of(d);
89  
90  					return new Info(mmsi, imo, cls, shipType, maxDraftMetres,
91  							dimensionAMetres, dimensionBMetres,
92  							dimensionCMetres, dimensionDMetres);
93  				});
94  	}
