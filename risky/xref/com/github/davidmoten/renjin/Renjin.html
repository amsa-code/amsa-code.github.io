3   import java.io.InputStream;
4   import java.io.InputStreamReader;
5   import java.nio.charset.Charset;
6   
7   import javax.script.ScriptEngine;
8   import javax.script.ScriptEngineManager;
9   import javax.script.ScriptException;
10  
11  public class Renjin {
12  
13  	public static void execute(InputStream is) {
14  		// create a script engine manager:
15  		ScriptEngineManager manager = new ScriptEngineManager();
16  		// create a Renjin engine:
17  		ScriptEngine engine = manager.getEngineByName("Renjin");
18  		// check if the engine has loaded correctly:
19  		if (engine == null) {
20  			throw new RuntimeException(
21  					"Renjin Script Engine not found on the classpath.");
22  		}
23  		try {
24  			engine.eval(new InputStreamReader(is, Charset.forName("UTF-8")));
25  		} catch (ScriptException e) {
26  			throw new RuntimeException(e);
27  		}
28  	}
29  
30  }
