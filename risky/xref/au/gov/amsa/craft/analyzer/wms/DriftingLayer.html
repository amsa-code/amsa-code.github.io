58      public DriftingLayer() {
59  
60          // collect drifting candidates
61  
62          // use these filenames as input NMEA
63          Observable<String> filenames = getFilenames();
64  
65          Observable<VesselPosition> aisPositions = getPositions(filenames)
66          // log
67                  .lift(Logging.<VesselPosition> logger().showCount().every(100000).log());
68  
69          new DriftingDetector().getCandidates(aisPositions)
70          // group by id and date
71                  .distinct(byIdAndDay())
72                  // add to queue
73                  .doOnNext(addToQueue())
74                  // run in background
75                  .subscribeOn(Schedulers.io())
76                  // subscribe
77                  .subscribe(createObserver());
78      }

80      private static Observable<VesselPosition> getPositions(Observable<String> filenames) {
81          return filenames
82          // get the positions from each file
83          // use concatMap till merge bug is fixed RxJava
84          // https://github.com/ReactiveX/RxJava/issues/1941
85          // log filename
86          // .lift(Logging.<String>
87          // logger().onNextPrefix("loading file=").showValue().log())
88          // extract positions from file
89                  .flatMap(filenameToPositions())
90                  // log
91                  // .lift(Logging.<VesselPosition>logger().log())
92                  // only class A vessels
93                  .filter(onlyClassA())
94                  // ignore vessels at anchor
95                  .filter(notAtAnchor())
96                  // is a big vessel
97                  .filter(isBig());

100     private static Observable<String> getFilenames() {
101         List<String> filenames = new ArrayList<String>();
102         final String filenameBase = "/media/analysis/nmea/2014/sorted-NMEA_ITU_201407";
103         for (int i = 1; i <= 31; i++) {
104             String filename = filenameBase + new DecimalFormat("00").format(i) + ".gz";
105             if (new File(filename).exists()) {
106                 filenames.add(filename);
107                 log.info("adding filename " + filename);
108             }
109         }
110         return Observable.from(filenames);

221     @Override
222     public String getInfo(Date time, WmsRequest request, final Point point, String mimeType) {
223 
224         final int HOTSPOT_SIZE = 5;
225 
226         final Projector projector = WmsUtil.getProjector(request);
227         final StringBuilder response = new StringBuilder();
228         response.append("<html>");
229         Observable.from(queue)
230         // only vessel positions close to the click point
231                 .filter(new Func1<VesselPosition, Boolean>() {
232                     @Override
233                     public Boolean call(VesselPosition p) {
234                         Point pt = projector.toPoint(p.lat(), p.lon());
235                         return Math.abs(point.x - pt.x) <= HOTSPOT_SIZE
236                                 && Math.abs(point.y - pt.y) <= HOTSPOT_SIZE;
237                     }
238                 })
239                 // add html fragment for each vessel position to the response
240                 .doOnNext(new Action1<VesselPosition>() {
241                     @Override
242                     public void call(VesselPosition p) {
243                         response.append("<p>");
244                         response.append("<a href=\"https://www.fleetmon.com/en/vessels?s="
245                                 + p.id().uniqueId() + "\">mmsi=" + p.id().uniqueId()
246                                 + "</a>, time=" + new Date(p.time()));
247                         if (p.shipType().isPresent()) {
248                             response.append(", ");
249                             response.append(ShipTypeDecoder.getShipType(p.shipType().get()));
250                         }
251                         response.append("</p>");
252                     }
253                 })
254                 // go!
255                 .subscribe();
256         response.append("</html>");
257         return response.toString();
258     }

274     private static void sortFile(String filename) throws FileNotFoundException, IOException {
275         Comparator<LineAndTime> comparator = new Comparator<LineAndTime>() {
276             @Override
277             public int compare(LineAndTime line1, LineAndTime line2) {
278                 return ((Long) line1.getTime()).compareTo(line2.getTime());
279             }
280         };
281         final File in = new File(filename);
282         final File outFile = new File(in.getParentFile(), "sorted-" + in.getName());
283         if (outFile.exists()) {
284             log.info("file exists: " + outFile);
285             return;
286         }
287         final OutputStreamWriter out = new OutputStreamWriter(new GZIPOutputStream(
288                 new FileOutputStream(outFile)), StandardCharsets.UTF_8);
289 
290         Streams
291         // read from file
292         .nmeaFromGzip(filename)
293         // get time
294                 .flatMap(Streams.toLineAndTime())
295                 // sort
296                 .lift(new SortOperator<LineAndTime>(comparator, 20000000))
297                 // .lift(Logging.<LineAndTime> logger().showValue().log())
298                 .doOnCompleted(new Action0() {
299                     @Override
300                     public void call() {
301                         try {
302                             out.close();
303                         } catch (IOException e) {
304                         }
305                     }
306                 }).forEach(new Action1<LineAndTime>() {
307                     @Override
308                     public void call(LineAndTime line) {
309                         try {
310                             out.write(line.getLine());
311                             out.write('\n');
312                         } catch (IOException e) {
313                             throw new RuntimeException(e);
314                         }
315                     }
316                 });
317     }

319     private static void sortFiles() throws FileNotFoundException, IOException {
320         // String filename = "/media/analysis/nmea/2014/NMEA_ITU_20140701.gz";
321         File directory = new File("/media/analysis/nmea/2014");
322         Preconditions.checkArgument(directory.exists());
323         File[] files = directory.listFiles(new FileFilter() {
324             @Override
325             public boolean accept(File f) {
326                 return f.getName().startsWith("NMEA_") && f.getName().endsWith(".gz");
327             }
328         });
329         int count = 0;
330 
331         Arrays.sort(files, new Comparator<File>() {
332             @Override
333             public int compare(File f1, File f2) {
334                 return f1.getPath().compareTo(f2.getPath());
335             }
336         });
337 
338         for (File file : files) {
339             count++;
340             log.info("sorting " + count + " of " + files.length + ": " + file);
341             sortFile(file.getAbsolutePath());
342         }
343     }
