16  public class Parquet {
17  
18      public static void writeTo(Iterable<GenericData.Record> recordsToWrite, Path path)
19              throws IOException {
20          writeTo(recordsToWrite, path, FIX_WITH_MMSI_SCHEMA);
21      }
22      
23      private static void writeTo(Iterable<GenericData.Record> recordsToWrite, Path path, Schema schema)
24              throws IOException {
25          try (ParquetWriter<GenericData.Record> writer = AvroParquetWriter.<GenericData.Record>builder(path)
26                  .withSchema(schema).withConf(new Configuration()).withCompressionCodec(CompressionCodecName.SNAPPY)
27                  .build()) {
28  
29              for (GenericData.Record record : recordsToWrite) {
30                  writer.write(record);
31              }
32          }
33      }
34  
35      public static final Schema FIX_WITH_MMSI_SCHEMA = loadSchema();
36  
37      public static GenericData.Record toRecord(Fix fix) {
38          GenericData.Record record = new GenericData.Record(FIX_WITH_MMSI_SCHEMA);
39          record.put("mmsi", fix.mmsi());
40          record.put("lat", fix.lat());
41          record.put("lon", fix.lon());
42          record.put("timeEpochMs", fix.time());
43          record.put("latencySeconds", fix.latencySeconds().or(-1));
44          record.put("source", (int) fix.source().or((short) 0));
45          record.put("navigationalStatus", fix.navigationalStatus().transform(n -> n.ordinal()).or(127));
46          record.put("rateOfTurn", fix.rateOfTurn().or((byte) -128));
47          record.put("speedOverGroundKnots", fix.speedOverGroundKnots().or(-1f));
48          record.put("courseOverGroundDegrees", fix.courseOverGroundDegrees().or(-1f));
49          record.put("headingDegrees", fix.headingDegrees().or(-1f));
50          record.put("aisClass", fix.aisClass() == AisClass.A);
51          return record;
52      }
53  
54      
55      @VisibleForTesting
56      static Schema loadSchema() {
57          try {
58              return new Schema.Parser().parse(Parquet.class.getResourceAsStream("/fixes.avsc"));
59          } catch (IOException e) {
60              throw new RuntimeException(e);
61          }
62      }
