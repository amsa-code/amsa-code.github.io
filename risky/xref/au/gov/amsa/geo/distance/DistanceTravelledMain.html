24  	private static Logger log = Logger.getLogger(DistanceTravelledMain.class);

26  	private static void run(String directory, Options options, boolean gui) {
27  
28  		final Observable<File> files = Util.getFiles(directory, "craft-");
29  
30  		CalculationResult result = calculateTrafficDensity(options, files, 1, 1);
31  
32  		if (gui) {
33  			DisplayPanel.displayGui(files, options, result);
34  		}
35  		String filename = result.getCells()
36  		// to bytes
37  				.lift(new OperatorCellValuesToBytes(options))
38  				// save bytes to a file
39  				.lift(new OperatorWriteBytes())
40  				// get filename
41  				.toBlocking().single();
42  		log.info("result saved to file " + filename);
43  
44  		CalculationResult resultFromFile = new CalculationResult(
45  				BinaryCellValuesObservable.readValues(new File(filename))
46  						.skip(1).cast(CellValue.class), result.getMetrics());
47  		// 8:5 is ok ratio
48  		saveAsPng(Renderer.createImage(options, 2, 1280, resultFromFile),
49  				new File("target/map.png"));
50  
51  		// saveCalculationResultAsText(options, result, "target/densities.txt");
52  		// saveCalculationResultAsBinary(options, result,
53  		// "target/densities.bin");
54  	}
55  
56  	private static Options createOptions(double cellSizeDegrees) {
57  		return Options.builder()
58  		// set origin latitude
59  				.originLat(0)
60  				// set origin longitudue
61  				.originLon(0)
62  				// square cell size in degrees
63  				.cellSizeDegrees(cellSizeDegrees)
64  				// set bounds
65  				.bounds(new Bounds(15, 67, -60, 179))
66  				// sabine bounds:
67  				// .bounds(new Bounds(-10, 110, -45, 158))
68  				// .bounds(new Bounds(15, 90, -20, 125))
69  				// set start
70  				// .startTime("2014-04-20")
71  				// set finish
72  				// .finishTime("2014-06-06")
73  				// set segment options
74  				.segmentOptions(SegmentOptions.builder()
75  				// set max speed knots
76  						.maxSpeedKnots(1000)
77  						// set max time per segment
78  						.maxTimePerSegmentMs(TimeUnit.DAYS.toMillis(1))
79  						// build
80  						.build())
81  				// build options
82  				.build();
83  	}
84  
85  	public static void main(String[] args) throws InterruptedException {
86  		log.info("starting");
87  		String directory;
88  		if (args.length > 0)
89  			directory = args[0];
90  		else
91  			directory = "/media/analysis/positions-365-days-2";
92  		// directory = System.getProperty("user.home")
93  		// + "/Downloads/positions-183-days";
94  		final double cellSizeDegrees;
95  		if (args.length > 1)
96  			cellSizeDegrees = Double.parseDouble(args[1]);
97  		else
98  			cellSizeDegrees = 0.5;
99  
100 		final Options options = createOptions(cellSizeDegrees);
101 		run(directory, options, false);
102 	}
