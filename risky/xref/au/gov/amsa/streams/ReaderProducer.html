1   package au.gov.amsa.streams;
2   
3   import java.io.IOException;
4   import java.io.Reader;
5   import java.util.Arrays;
6   import java.util.concurrent.atomic.AtomicLong;
7   
8   import rx.Producer;
9   import rx.Subscriber;
10  
11  class ReaderProducer implements Producer {
12  
13  	private final AtomicLong requested = new AtomicLong(0);
14  	private final Reader reader;
15  	private final Subscriber<? super String> subscriber;
16  	private final int size;
17  
18  	ReaderProducer(Reader reader, Subscriber<? super String> subscriber,
19  			int size) {
20  		this.reader = reader;
21  		this.subscriber = subscriber;
22  		this.size = size;
23  	}
24  
25  	@Override
26  	public void request(long n) {
27  		try {
28  			if (requested.get() == Long.MAX_VALUE)
29  				// already started with fast path
30  				return;
31  			else if (n == Long.MAX_VALUE) {
32  				// fast path
33  				requestAll();
34  			} else
35  				requestSome(n);
36  		} catch (RuntimeException e) {
37  			subscriber.onError(e);
38  		} catch (IOException e) {
39  			subscriber.onError(e);
40  		}
41  	}
42  
43  	private void requestAll() {
44  		requested.set(Long.MAX_VALUE);
45  		char[] buffer = new char[size];
46  		try {
47  			if (subscriber.isUnsubscribed())
48  				return;
49  			int n = reader.read(buffer);
50  			while (n != -1 && !subscriber.isUnsubscribed()) {
51  				subscriber.onNext(new String(Arrays.copyOf(buffer, n)));
52  				if (!subscriber.isUnsubscribed())
53  					n = reader.read(buffer);
54  			}
55  		} catch (IOException e) {
56  			subscriber.onError(e);
57  		}
58  		if (subscriber.isUnsubscribed())
59  			return;
60  		subscriber.onCompleted();
61  	}
62  
63  	private void requestSome(long n) throws IOException {
64  		// back pressure path
65  		// this algorithm copied roughly from
66  		// rxjava/OnSubscribeFromIterable.java
67  
68  		long previousCount = requested.getAndAdd(n);
69  		if (previousCount == 0) {
70  			char[] buffer = new char[size];
71  			while (true) {
72  				long r = requested.get();
73  				long numToEmit = r;
74  
75  				// emit numToEmit
76  
77  				if (subscriber.isUnsubscribed())
78  					return;
79  				int numRead;
80  				if (numToEmit > 0)
81  					numRead = reader.read(buffer);
82  				else
83  					numRead = 0;
84  				while (numRead != -1 && !subscriber.isUnsubscribed()
85  						&& numToEmit > 0) {
86  					subscriber
87  							.onNext(new String(Arrays.copyOf(buffer, numRead)));
88  					numToEmit--;
89  					if (numToEmit > 0 && !subscriber.isUnsubscribed())
90  						numRead = reader.read(buffer);
91  				}
92  
93  				// check if we have finished
94  				if (numRead == -1 && !subscriber.isUnsubscribed())
95  					subscriber.onCompleted();
96  				else if (subscriber.isUnsubscribed())
97  					return;
98  				else if (requested.addAndGet(-r) == 0)
99  					return;
100 			}
101 		}
102 	}
103 }
