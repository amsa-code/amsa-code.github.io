29      /**
30       * Returns an Observable sequence of lines from the given host and port. If
31       * the stream is quiet for <code>quietTimeoutMs</code> then a reconnect will
32       * be attempted. Note that this is a good idea with TCPIP connections as for
33       * instance a firewall can simply drop a quiet connection without the client
34       * being aware of it. If any exception occurs a reconnect will be attempted
35       * after <code>reconnectDelayMs</code>. If the socket is closed by the
36       * server (the end of the input stream is reached) then a reconnect is
37       * attempted after <code>reconnectDelayMs</code>.
38       * 
39       * @param hostPort
40       * @return Observable sequence of lines (including possible new line
41       *         character at end if exists).
42       */
43      public static Observable<String> from(final String host, final int port, long quietTimeoutMs,
44              long reconnectDelayMs, Charset charset) {
45          // delay connect by delayMs so that if server closes
46          // stream on every connect we won't be in a mad loop of
47          // failing connections
48          return triggersWithDelay(reconnectDelayMs)
49          // connect to server and read lines from its input stream
50                  .concatMap(from(host, port, charset, quietTimeoutMs))
51                  // ensure connection has not dropped out by throwing an
52                  // exception after a minute of no messages. This is a good idea
53                  // with TCPIP because for example a firewall might drop a quiet
54                  // connection and we won't know about it. The from method call
55                  // in concatMap above sets a socket read timeout as well but
56                  // just in case we pull the plug on the socket a second later
57                  // with this call to timeout()
58                  .timeout(quietTimeoutMs + 1000, TimeUnit.MILLISECONDS)
59                  // if any exception occurs retry
60                  .retry()
61                  // all subscribers use the same stream
62                  .share();
63      }

122     /**
123      * Returns a <i>synchronous</i>s stream of a count of integers with delayMs
124      * between emissions.
125      * 
126      * @param delayMs
127      * @return stream of integers starting at 1 with delayMs between emissions
128      */
129     private static Observable<Integer> triggersWithDelay(long delayMs) {
130         return just(1)
131         // keep counting
132                 .concatWith(
133                 // numbers from 2 on now with delay
134                         range(2, Integer.MAX_VALUE - 1)
135                         // delay till next number released
136                                 .delay(delayMs, TimeUnit.MILLISECONDS, Schedulers.immediate()));
137     }

139     private static Func1<Integer, Observable<String>> from(final String host, final int port,
140             final Charset charset, long quietTimeoutMs) {
141         return new Func1<Integer, Observable<String>>() {
142             @Override
143             public Observable<String> call(Integer n) {
144                 return Observable
145                 // create a stream from a socket and dispose of socket
146                 // appropriately
147                         .using(socketCreator(host, port, quietTimeoutMs),
148                                 socketObservableFactory(charset), socketDisposer(), true)
149                         // cannot ask host to slow down so buffer on
150                         // backpressure
151                         .onBackpressureBuffer();
152             }
153         };
154     }

156     @VisibleForTesting
157     static Func0<Socket> socketCreator(final String host, final int port, long quietTimeoutMs) {
158         return Checked.f0(() -> {
159             log.info("creating socket to " + host + ":" + port);
160             Socket socket = new Socket(host, port);
161             socket.setSoTimeout((int) quietTimeoutMs);
162             return socket;
163         });
164     }

166     @VisibleForTesting
167     static Func1<Socket, Observable<String>> socketObservableFactory(final Charset charset) {
168         return Checked.f1(socket -> Strings.from(new InputStreamReader(socket.getInputStream(),
169                 charset)));
170     }
171 
172     @VisibleForTesting
173     static Action1<Socket> socketDisposer() {
174         return socket -> {
175             try {
176                 log.info("closing socket " + socket.getInetAddress().getHostAddress() + ":"
177                         + socket.getPort());
178                 socket.close();
179             } catch (IOException e) {
180                 // don't really care if socket could not be closed cleanly
181                 log.info("messageOnSocketClose=" + e.getMessage(), e);
182             }
183         };
184     }

186     public static Observable<String> mergeLinesFrom(Observable<HostPort> hostPorts) {
187         return hostPorts
188         //
189                 .map(hp -> StringSockets
190                         .from(hp.getHost(), hp.getPort(), hp.getQuietTimeoutMs(),
191                                 hp.getReconnectDelayMs(), StandardCharsets.UTF_8)
192                         // split by new line character
193                         .compose(o -> Strings.split(o, "\n"))
194                         // make asynchronous
195                         .subscribeOn(Schedulers.io()))
196                 // merge streams of lines
197                 .compose(o -> Observable.merge(o));
198     }
