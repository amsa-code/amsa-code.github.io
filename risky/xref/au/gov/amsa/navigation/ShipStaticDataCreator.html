34      public static Observable<AisShipStatic> writeStaticDataToFile(List<File> files, File outputFile,
35              Scheduler scheduler) {
36          Func0<PrintStream> resourceFactory = Checked.f0(() -> new PrintStream(outputFile));
37          Func1<PrintStream, Observable<AisShipStatic>> observableFactory = out -> Observable
38                  .from(files)
39                  // buffer into chunks for each processor
40                  .buffer(Math.max(1,
41                          files.size()
42                                  / Runtime.getRuntime().availableProcessors())
43                          - 1)
44                  .flatMap(
45                          list -> Observable.from(list) //
46                                  .lift(Logging.<File> logger().showValue().showMemory().log()) //
47                                  .concatMap(
48                                          file -> Streams.extract(Streams.nmeaFromGzip(file)) //
49                                                  .flatMap(aisShipStaticOnly) //
50                                                  .map(m -> m.getMessage().get().message()) //
51                                                  .filter(m -> m instanceof AisShipStatic) //
52                                                  .cast(AisShipStatic.class) //
53                                                  .distinct(m -> m.getMmsi()) //
54                                                  .doOnError(e -> System.err.println("could not read "
55                                                          + file + ": " + e.getMessage())) //
56                                  .onErrorResumeNext(Observable.<AisShipStatic> empty())) //
57                          .distinct(m -> m.getMmsi()) //
58                          .subscribeOn(scheduler)) //
59                  .distinct(m -> m.getMmsi()) //
60                  .compose(Transformers.mapWithIndex()) //
61                  .doOnNext(indexed -> {
62                      if (indexed.index() == 0) {
63                          out.println(
64                                  "# MMSI, IMO, AisClass, AisShipType, MaxPresentStaticDraughtMetres, DimAMetres, DimBMetres, DimCMetres, DimDMetres, LengthMetres, WidthMetres, Name");
65                          out.println("# columns are tab delimited");
66                          out.println("# -1 = not present");
67                      }
68                  })
69                  //
70                  .map(indexed -> indexed.value())
71                  //
72                  .doOnNext(m -> {
73                      out.format("%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\n", m.getMmsi(),
74                              getImo(m).or(-1), m instanceof AisShipStaticA ? "A" : "B",
75                              m.getShipType(), getMaximumPresentStaticDraughtMetres(m).or(-1F),
76                              m.getDimensionA().or(-1), m.getDimensionB().or(-1),
77                              m.getDimensionC().or(-1), m.getDimensionD().or(-1),
78                              AisShipStaticUtil.lengthMetres(m).or(-1),
79                              AisShipStaticUtil.widthMetres(m).or(-1), prepareName(m.getName()));
80                      out.flush();
81                  });
82  
83          Action1<PrintStream> disposeAction = out -> out.close();
84          return Observable.using(resourceFactory, observableFactory, disposeAction);
85      }
86  
87      private static String prepareName(String name) {
88          if (name == null)
89              return "";
90          else
91              return name.replace("\t", " ").trim();
92      }
93  
94      private static Optional<Integer> getImo(AisShipStatic m) {
95          if (m instanceof AisShipStaticA) {
96              return ((AisShipStaticA) m).getImo();
97          } else
98              return Optional.absent();
99      }
100 
101     private static Optional<Float> getMaximumPresentStaticDraughtMetres(AisShipStatic m) {
102         if (m instanceof AisShipStaticA) {
103             return Optional.of((float) ((AisShipStaticA) m).getMaximumPresentStaticDraughtMetres());
104         } else
105             return Optional.absent();
106     }

108     private static Func1<TimestampedAndLine<AisMessage>, Observable<TimestampedAndLine<AisShipStatic>>> aisShipStaticOnly = m -> {
109         Optional<Timestamped<AisMessage>> message = m.getMessage();
110         if (message.isPresent() && message.get().message() instanceof AisShipStatic) {
111             @SuppressWarnings("unchecked")
112             TimestampedAndLine<AisShipStatic> m2 = (TimestampedAndLine<AisShipStatic>) (TimestampedAndLine<?>) m;
113             return Observable.just(m2);
114         } else
115             return Observable.empty();
116     };
