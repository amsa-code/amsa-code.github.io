1   package au.gov.amsa.navigation;
2   
3   import rx.Observable;
4   import rx.Observable.Operator;
5   import rx.Observable.Transformer;
6   import au.gov.amsa.navigation.DriftDetectorOperator.Options;
7   import au.gov.amsa.risky.format.HasFix;
8   
9   import com.github.davidmoten.rx.Operators;
10  
11  public class DriftDetector {
12  
13  	public Observable<DriftCandidate> getCandidates(Observable<HasFix> o,
14  			Options options) {
15  		return o.lift(detectDriftCandidates(options));
16  	}
17  
18  	public static DriftDetectorTransformer detectDrift() {
19  		return new DriftDetectorTransformer(Options.instance());
20  	}
21  
22  	/**
23  	 * This operator expects a stream of fixes of increasing time except when
24  	 * the mmsi changes (it can!).
25  	 * 
26  	 * @return an operator to detect drift candidates
27  	 */
28  	private static Operator<DriftCandidate, HasFix> detectDriftCandidates(
29  			Options options) {
30  		return new DriftDetectorOperator(options);
31  	}
32  
33  	public static DriftDetectorTransformer detectDrift(Options options) {
34  		return new DriftDetectorTransformer(options);
35  	}
36  
37  	public static class DriftDetectorTransformer implements
38  			Transformer<HasFix, DriftCandidate> {
39  
40  		private final DriftDetector d = new DriftDetector();
41  		private final Options options;
42  
43  		public DriftDetectorTransformer(Options options) {
44  			this.options = options;
45  		}
46  
47  		@Override
48  		public Observable<DriftCandidate> call(Observable<HasFix> o) {
49  			return d.getCandidates(o, options)
50  					.lift(Operators.bufferEmissions());
51  		}
52  	}
53  
54  }
