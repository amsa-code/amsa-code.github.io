12      public static List<HexAttribute> coarsePositionAttributes(double latSeconds, double lonSeconds,
13              int s, int f) {
14          List<HexAttribute> list = new ArrayList<>();
15          list.add(new HexAttribute(AttributeType.COARSE_POSITION_LATITUDE, s, f,
16                  latSeconds / 3600.0 + "", ""));
17          list.add(new HexAttribute(AttributeType.COARSE_POSITION_LONGITUDE, s, f,
18                  lonSeconds / 3600.0 + "", ""));
19          return list;
20      }

32      static HexAttribute mmsiFromBinary(AttributeType type, String binCode, int s, int f) {
33          int v = Conversions.binaryToDecimal(binCode.substring(s, f + 1));
34          String e = "";
35          String value = v + "";
36          int len = value.length();
37          for (int i = 0; i < (6 - len); i++) {
38              value = "0" + value;
39          }
40          return new HexAttribute(type, s, f, value, e);
41      }
