8   class UserOrbitography extends User {
9   
10      UserOrbitography() {
11          super("Orbitography", "000");
12      }
13  
14      @Override
15      List<HexAttribute> decode(String hexStr) {
16          String binCode = Conversions.hexToBinary(hexStr);
17  
18          List<HexAttribute> result = new ArrayList<HexAttribute>();
19  
20          userFragment(this, hexStr, binCode, result);
21  
22          result = this.orbitographyData(result, binCode, 40, 85);
23  
24          // The if condition is always true because it is prefixed to be 30 chars already
25          if (hexStr.length() > 15) {
26              result.add(this.bch1(binCode, 86, 106));
27  
28              if (this.isLongMessage(binCode)) {
29                  result.add(this.nationalUse(binCode, 107, 144));
30              } else if (binCode.charAt(107) == '0') {
31                  result.add(this.nationalUse(binCode, 107, 112));
32              } else {
33                  result.add(new HexAttribute(AttributeType.EMERGENCY_CODE_FLAG, 107,
34                          "Non-Spec (" + binCode.charAt(107) + ")", ""));
35                  result.add(this.nationalUse(binCode, 108, 112));
36              }
37          }
38  
39          return result;
40      }
41  
42      // This is where changes go
43      private List<HexAttribute> orbitographyData(List<HexAttribute> result, String binCode, int s,
44              int f) {
45          String v1 = binCode.substring(s, f + 1);
46          String e1 = "";
47  
48          result.add(new HexAttribute(AttributeType.ORBITOGRAPHY_DATA, s, f, v1, e1));
49  
50          // bit 40-81
51          String[] vE = Conversions.mBaudotBits2mBaudotStr(this.getName(),
52                  binCode.substring(s, f - 4 + 1), 6);
53          // 24 June 2005, double quote is not used any more
54          // String v2 = "\"" + vE[0] + "\"";
55          String v2 = vE[0];
56          String e2 = vE[1];
57  
58          if (e2 != null && e2.length() > 0) {
59              e2 = "\nWARNING - SUSPECT NON-SPEC IN ORBITOGRAPHY BEACON CLEAR TEXT IDENTIFIER\n" + e2;
60          }
61  
62          result.add(new HexAttribute(AttributeType.ORBITOGRAPHY_ID, s, f - 4, v2, e2));
63  
64          // Bit 82-85
65          String v3 = binCode.substring(f - 4 + 1, f + 1);
66  
67          if (v3.equals("0000")) {
68              result.add(new HexAttribute(AttributeType.ORBITOGRAPHY_4_BITS, f - 4 + 1, f, v3, ""));
69          } else {
70              result.add(new HexAttribute(AttributeType.ORBITOGRAPHY_4_BITS, f - 4 + 1, f,
71                      v3 + " (Non-Spec)", ""));
72          }
73  
74          return result;
75      }
