12      @Override
13      public HexAttribute aircraftOperator(String binCode, int s, int f) {
14          // 11 May 2005
15          String[] vE = Conversions.mBaudotBits2mBaudotStr(this.getName() + " " + serialBeaconType(),
16                  binCode.substring(s, f + 1), 6);
17          String v = vE[0];
18          String e = vE[1];
19  
20          return new HexAttribute(AttributeType.AIRCRAFT_OPERATOR, s, f, v, e);
21      }
