3   class StandardLocationShipSecurityAlertSystem extends StandardLocationBase {
4   
5       StandardLocationShipSecurityAlertSystem() {
6           super("1100", "Ship Security MMSI", (s, binCode, result) -> {
7               result.add(Util.mmsiFromBinary(AttributeType.SHIP_SECURITY_MMSI, binCode, 41, 60));
8               result.add(((StandardLocationShipSecurityAlertSystem) s).securityFixedBits(binCode, 61,
9                       64));
10          });
11      }
12  
13      private HexAttribute securityFixedBits(String binCode, int s, int f) {
14          String v = binCode.substring(s, f + 1);
15          String e = "";
16  
17          // 08 Aug 2006
18          if (!v.equals("0000")) {
19              v += " (Non-Spec)";
20          }
21  
22          return new HexAttribute(AttributeType.FIXED_BITS, s, f, v, e);
23      }
