16      // For bit 109-112 (Maritime Emergency Codes). See C/S T.001 Table A4
17      static List<HexAttribute> maritimeEmergencyCodes(List<HexAttribute> result, String binCode) {
18          // Convert bit 109-112 to decimal
19          int bitToNum = Conversions.binaryToDecimal(binCode.substring(109, 113));
20  
21          String mec = "Maritime Emergency Codes";
22          AttributeType name = AttributeType.NATURE_OF_DISTRESS;
23  
24          HexAttribute maritimeEmergencyCodes;
25  
26          switch (bitToNum) {
27          case 0:
28              maritimeEmergencyCodes = new HexAttribute(name, 109, 112,
29                      mec + "(Unspecified distress) " + binCode.substring(109, 113), "");
30              break;
31          case 1:
32              maritimeEmergencyCodes = new HexAttribute(name, 109, 112,
33                      mec + "(Fire/explosion) " + binCode.substring(109, 113), "");
34              break;
35          case 2:
36              maritimeEmergencyCodes = new HexAttribute(name, 109, 112,
37                      mec + "(Flooding) " + binCode.substring(109, 113), "");
38              break;
39          case 3:
40              maritimeEmergencyCodes = new HexAttribute(name, 109, 112,
41                      mec + "(Collision) " + binCode.substring(109, 113), "");
42              break;
43          case 4:
44              maritimeEmergencyCodes = new HexAttribute(name, 109, 112,
45                      mec + "(Grounding) " + binCode.substring(109, 113), "");
46              break;
47          case 5:
48              maritimeEmergencyCodes = new HexAttribute(name, 109, 112,
49                      mec + "(Listing, in danger of capsizing) " + binCode.substring(109, 113), "");
50              break;
51          case 6:
52              maritimeEmergencyCodes = new HexAttribute(name, 109, 112,
53                      mec + "(Sinking) " + binCode.substring(109, 113), "");
54              break;
55          case 7:
56              maritimeEmergencyCodes = new HexAttribute(name, 109, 112,
57                      mec + "(Disabled and adrift) " + binCode.substring(109, 113), "");
58              break;
59          case 8:
60              maritimeEmergencyCodes = new HexAttribute(name, 109, 112,
61                      mec + "(Abandoning ship) " + binCode.substring(109, 113), "");
62              break;
63          default:
64              maritimeEmergencyCodes = new HexAttribute(name, 109, 112,
65                      mec + "(spare) " + binCode.substring(109, 113), "");
66          }
67  
68          result.add(maritimeEmergencyCodes);
69  
70          return result;
71      }

73      private static HexAttribute fireFlag(String binCode, int s) {
74          String v = "";
75          String e = "";
76          if (binCode.charAt(s) == '1') {
77              v = "YES";
78          } else {
79              v = "NO";
80          }
81  
82          // This is grouped under "Nature of Distress" with the identation of 30 white
83          // spaces
84          // return new HexAttribute(" ", s, v, e);
85          return new HexAttribute(AttributeType.FIRE_FLAG, s, v, e);
86      }

113     private static HexAttribute emergencySpare(String binCode, int s) {
114         String v = "";
115         String e = "";
116         if (binCode.charAt(s) == '0') {
117             v = binCode.charAt(s) + " (N/A)";
118         } else {
119             v = binCode.charAt(s) + " (Non-Spec)";
120         }
121 
122         return new HexAttribute(AttributeType.SPARE, s, v, e);
123     }

126     static List<HexAttribute> nonMaritimeEmergencyCodes(List<HexAttribute> result, String binCode) {
127         result.add(new HexAttribute(AttributeType.NATURE_OF_DISTRESS, 109, 112,
128                 binCode.substring(109, 113), ""));
129         result.add(fireFlag(binCode, 109));
130         result.add(medicalHelpFlag(binCode, 110));
131         result.add(disabledFlag(binCode, 111));
132         result.add(emergencySpare(binCode, 112));
133         return result;
134     }

165     /**
166      * @param binCode        input binary string
167      * @param start          start index
168      * @param length         length of position encoding
169      * @param secondsPerUnit factor to use when converting a decimal number to
170      *                       seconds
171      * @return position
172      */
173     static Position position(String binCode, int start, int length, int secondsPerUnit) {
174         final int lonLength;
175         // if odd number of bits then lon gets one more than lat
176         if (length % 2 != 0) {
177             lonLength = floorDiv(length, 2) + 1;
178         } else {
179             lonLength = length / 2;
180         }
181         final int latLength = length - lonLength;
182 
183         final int latSeconds = extractLatSeconds(binCode, start, secondsPerUnit, latLength);
184         final int lonSeconds = extractLonSeconds(binCode, start, secondsPerUnit, lonLength, latLength);
185         return new Position(latSeconds, lonSeconds);
186     }
187 
188     private static int extractLonSeconds(String binCode, int start, int secondsPerUnit,
189             final int lonLength, final int latLength) {
190         String lonBits = binCode.substring(start + latLength + 1, start + latLength + lonLength);
191         int code = Conversions.binaryToDecimal(lonBits);
192         final int codeSeconds = code * secondsPerUnit;
193         final int deg = codeSeconds / 3600;
194         final int min = codeSeconds % 3600 / 60;
195         int seconds = deg * 3600 + min * 60;
196 
197         if (binCode.charAt(start + latLength) == '1') {
198             seconds = seconds * -1;
199         }
200         return seconds;
201     }
202 
203     private static int extractLatSeconds(String binCode, int start, int secondsPerUnit,
204             final int latLength) {
205         String latBits = binCode.substring(start + 1, start + latLength);
206         int code = Conversions.binaryToDecimal(latBits);
207         final int codeSeconds = code * secondsPerUnit;
208         final int deg = codeSeconds / 3600;
209         final int min = codeSeconds % 3600 / 60;
210         int seconds = deg * 3600 + min * 60;
211         if (binCode.charAt(start) == '1') {
212             seconds = seconds * -1;
213         }
214         return seconds;
215     }

217     /**
218      * Returns the largest (closest to positive infinity) {@code int} value that is
219      * less than or equal to the algebraic quotient. There is one special case, if
220      * the dividend is the {@linkplain Integer#MIN_VALUE Integer.MIN_VALUE} and the
221      * divisor is {@code -1}, then integer overflow occurs and the result is equal
222      * to the {@code Integer.MIN_VALUE}.
223      * <p>
224      * Normal integer division operates under the round to zero rounding mode
225      * (truncation). This operation instead acts under the round toward negative
226      * infinity (floor) rounding mode. The floor rounding mode gives different
227      * results than truncation when the exact result is negative.
228      * <ul>
229      * <li>If the signs of the arguments are the same, the results of
230      * {@code floorDiv} and the {@code /} operator are the same. <br>
231      * For example, {@code floorDiv(4, 3) == 1} and {@code (4 / 3) == 1}.</li>
232      * <li>If the signs of the arguments are different, the quotient is negative and
233      * {@code floorDiv} returns the integer less than or equal to the quotient and
234      * the {@code /} operator returns the integer closest to zero.<br>
235      * For example, {@code floorDiv(-4, 3) == -2}, whereas {@code (-4 / 3) == -1}.
236      * </li>
237      * </ul>
238      * <p>
239      *
240      * @param x the dividend
241      * @param y the divisor
242      * @return the largest (closest to positive infinity) {@code int} value that is
243      *         less than or equal to the algebraic quotient.
244      * @throws ArithmeticException if the divisor {@code y} is zero
245      * @see #floorMod(int, int)
246      * @see #floor(double)
247      * @since 1.8
248      */
249     // copied from java 1.8
250     private static int floorDiv(int x, int y) {
251         int r = x / y;
252         // if the signs are different and modulo not zero, round down
253         if ((x ^ y) < 0 && (r * y != x)) {
254             r--;
255         }
256         return r;
257     }
