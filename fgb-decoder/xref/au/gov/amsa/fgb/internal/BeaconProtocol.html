146     private char rlsTacNumberPrefix(String binCode, int s) {
147         final String v = binCode.substring(s, s + 2);
148         if (v.equals("00")) {
149             return '2';
150         } else if (v.equals("01")) {
151             return '1';
152         } else if (v.equals("10")) {
153             return '3';
154         } else {
155             return 'T';
156         }

216     List<HexAttribute> aircraft24BitAddress(String binCode, int s, int f) {
217         String code = binCode.substring(s, f + 1);
218         String e = "";
219         List<HexAttribute> list = new ArrayList<>();
220         String hex = Conversions.binaryToHex(code);
221         String oct = removeLeadingZeros(Conversions.binaryToOctal(code));
222 
223         list.add(new HexAttribute(AttributeType.AIRCRAFT_24_BIT_ADDRESS_HEX, s, f, hex, e));
224         list.add(new HexAttribute(AttributeType.AIRCRAFT_24_BIT_ADDRESS_OCTAL, s, f, oct, e));
225 
226         HexAttribute h = aircraftCallSign(binCode, s, f);
227         String result = h.value();
228         if (result.length() > 0) {
229             // 13 May 2005
230             // Make the output align to the rest output
231             list.add(new HexAttribute(AttributeType.AIRCRAFT_CALL_SIGN, s, f, result, e));
232         }
233 
234         String countryReg = Conversions.getAircraftCountryOfReg(code);
235         if (countryReg.length() > 0) {
236             list.add(new HexAttribute(AttributeType.COUNTRY_OF_REGISTRATION, s, f, countryReg, e));
237         }
238         return list;
239     }

241     private static HexAttribute aircraftCallSign(String binCode, int s, int f) {
242         String e = "";
243         String v = "";
244         String country = binCode.substring(s, s + 6);
245 
246         // System.out.println("In BeaconProtocol.java, country=" + country);
247 
248         // if the first 6 bits are 011111 then it is an AUST reg beacon
249         // we can use an alg. to work out the callsign
250         if (country.equals("011111")) {
251             // 10 May 2005, based on the CASA algorithm passed by Scott
252             // Lillinton
253             // Discard the first 6 bits, only use the rest 18 bits
254             String code = binCode.substring(s + 6, f + 1);
255 
256             // String bin = binCode;
257             // String code = bin.substring(6, bin.length());
258 
259             v = Conversions.binaryToAircraftCallsign(code);
260         }
261 
262         // System.out.println("In BeaconProtocol.java, callSign: "+v);
263 
264         return new HexAttribute(AttributeType.AIRCRAFT_CALL_SIGN, s, f, v, e);
265     }

391     HexAttribute bch1(String binCodeOrig, int s, int f) {
392         // System.out.println("Orig " + binCodeOrig);
393         String binCode = binCodeOrig.substring(25, s);
394         String binCode2 = binCodeOrig.substring(s, f + 1);
395 
396         String e = "";
397 
398         binCode = addZerosOnRight(binCode, 61 - binCode.length());
399 
400         binCode = addZerosOnRight(binCode, 21);
401 
402         String bchCode = calcBCHCODE(binCode, GENERATOR_POLYNOMIAL);
403 
404         if (!bchCode.equals(binCode2)) {
405             e = "WARNING - ERROR IN FIRST PROTECTED FIELD\n";
406             e = e + " BCH SHOULD BE " + bchCode + "\n";
407             e = e + " BCH IS ...... " + binCode2;
408 
409             // 04/04/2005
410             // Add more words for BCH errors
411             e = "\n" + e + "\n\nNotes on Miscoded Beacons and BCH Errors:\n";
412             e += "1. Section 4.2.5.3 of C/S T.002 - LUT Specifications\n";
413             e += "2. Section III/B.1.1.3 of C/S A.001 - Data Distribution Plan\n\n";
414         }
415 
416         return new HexAttribute(AttributeType.ERROR_CORRECTING_CODE_1, s, f, binCode2, e);
417     }

420     // Overloaded method, called by StandardLocation's subclasses and
421     // NationalLocation's subclasses
422     List<HexAttribute> bch1(List<HexAttribute> result, String binCode, HexAttribute hexId) {
423         HexAttribute bch1 = bch1(binCode, 86, 106);
424 
425         String error = bch1.error();
426         if (error != null && error.length() > 0) {
427             // To remove the hexId with the default value in b66-85 if not the
428             // same
429             String hexIdDefaultLoc = hexId.value();
430             HexAttribute hexIdOrig = hexId(binCode, 26, 85);
431             String hexIdOrigLoc = hexIdOrig.value();
432 
433             if (!hexIdDefaultLoc.equalsIgnoreCase(hexIdOrigLoc)) {
434                 int hexIdInd = result.indexOf(hexId);
435                 result.remove(hexId);
436                 result.add(hexIdInd, hexIdOrig);
437 
438                 int noteInd = error.indexOf("\n\nNotes on Miscoded Beacons and BCH Errors:");
439                 String errPart1 = error.substring(0, noteInd);
440                 String errPart2 = error.substring(noteInd);
441 
442                 error = errPart1 + "\n Hex Id with default location: " + hexId.value() + "\n"
443                         + errPart2;
444                 bch1 = new HexAttribute(bch1.desc(), bch1.start(), bch1.finish() + "", error);
445             }
446         }
447 
448         result.add(bch1);
449         return result;
450     }

452     HexAttribute bch2(String binCodeOrig, int s, int f) {
453         String binCode = binCodeOrig.substring(107, s);
454         String binCode2 = binCodeOrig.substring(s, f + 1);
455         String e = "";
456 
457         binCode = addZerosOnRight(binCode, 26 - binCode.length());
458         binCode = addZerosOnRight(binCode, 12);
459 
460         String bchCode = calcBCHCODE(binCode, "1010100111001");
461 
462         if (!bchCode.equals(binCode2)) {
463             e = "WARNING - ERROR IN SECOND PROTECTED FIELD\n";
464             e = e + " BCH SHOULD BE " + bchCode + "\n";
465             e = e + " BCH IS ...... " + binCode2;
466         }
467 
468         return new HexAttribute(AttributeType.ERROR_CORRECTING_CODE_2, s, f, binCode2, e);
469     }

471     private static String addZerosOnRight(String s, int n) {
472         StringBuilder b = new StringBuilder(s);
473         for (int i = 0; i < n; i++) {
474             b.append("0");
475         }
476         return b.toString();
477     }

479     private static String calcBCHCODE(String bitCode, String generatorPolynomial) {
480 
481         int b = generatorPolynomial.length();
482         String result = bitCode.substring(0, b);
483         result = removeLeadingZeros(result);
484 
485         // Short protocols will have any leading 0 removed from the
486         // above call to removeLeadingZeros. So we should even up
487         // the ledger by removing the leading '0' from bitCode.
488         if (bitCode.charAt(0) == '0') {
489             bitCode = bitCode.substring(1, bitCode.length());
490         }
491 
492         for (int i = b - 1; i < bitCode.length(); i++) {
493             char c = bitCode.charAt(i);
494 
495             if (result.length() < b) {
496                 result += c;
497             }
498             if (result.length() == b) {
499                 result = xor(result, generatorPolynomial);
500             }
501         }
502 
503         while (result.length() < b - 1) {
504             result = "0" + result;
505         }
506 
507         return result;
508     }

510     private static String removeLeadingZeros(String str) {
511         int i = 0;
512         while (i < str.length() && str.charAt(i) == '0') {
513             i++;
514         }
515         return str.substring(i);
516     }

518     // do a bitwise XOR of G and R to form new R
519     private static String xor(String bitStrA, String bitStrB) {
520         StringBuilder result = new StringBuilder();
521         for (int i = 0; i < bitStrA.length(); i++) {
522             boolean bA = bitStrA.charAt(i) == '1';
523             boolean bB = bitStrB.charAt(i) == '1';
524             if (bA == bB) {
525                 result.append("0");
526             } else {
527                 result.append("1");
528             }
529         }
530         return removeLeadingZeros(result.toString());
531     }
