65      private Detection(Bits bits) {
66          Preconditions.checkArgument(bits.length() == 202, "length should be 202 but was " + bits.length());
67          this.bits = bits;
68          tac = bits.readUnsignedInt(16);
69          serialNo = bits.readUnsignedInt(14);
70          countryCode = bits.readUnsignedInt(10);
71          hasAtLeastOneEnabledHomingSignal = bits.readBoolean();
72          hasEnabledRls = bits.readBoolean();
73          testProtocolMessage = bits.readBoolean();
74          encodedGnssPosition = readPosition(bits.readBits(47));
75          vesselId = readVesselId(bits);
76          beaconType = readBeaconType(bits);
77          // skip spare bits
78          bits.skip(14);
79          rotatingField = readRotatingField(bits);
80          beacon23HexId = readBeacon23HexID(bits.position(0));
81          beacon15HexId = beacon23HexId.substring(0, 15);
82      }

98      private String readBeacon23HexID(Bits bits) {
99          StringBuilder s = new StringBuilder();
100         s.append("1");
101         bits.position(0);
102         String tacBits = bits.readBitString(16);
103         String serialBits = bits.readBitString(14);
104         String countryCodeBits = bits.readBitString(10);
105         s.append(countryCodeBits);
106         s.append("101");
107         s.append(tacBits);
108         s.append(serialBits);
109         String testProtocolFlagBits = bits.position(42).readBitString(1);
110         s.append(testProtocolFlagBits);
111         String vesselIdBits = bits.position(90).readBitString(47);
112         s.append(vesselIdBits);
113         Bits b = Bits.from(s.toString());
114         return b.readHex(23);
115     }

292     private static Rls readRotatingFieldRls(Bits bits) {
293         bits.skip(2);

296         bits.skip(4);

299         bits.skip(11);
300         return new Rls(canProcessAutomaticallyGeneratedAckRlmType1, canProcessManuallyGeneratedRlm, rlsProvider,
301                 beaconFeedback);

320     static Optional<BeaconFeedback> readBeaconFeadback(Bits bits, RlsProvider rlsProvider) {
321         boolean rlmType1FeedbackReceived = bits.readBoolean();
322         boolean rlmType2FeedbackReceived = bits.readBoolean();
323         RlsType rlsType = readRlsType(bits);
324         if (rlsProvider == RlsProvider.GALILEO) {
325             final Optional<String> shortRlmParametersBitString;
326             if (rlsType == RlsType.ACKNOWLEDGEMENT_SERVICE) {
327                 bits.skip(2);
328                 shortRlmParametersBitString = Optional.of(bits.readBitString(13));
329                 // skip parity bit
330                 bits.skip(1);
331             } else if (rlsType == RlsType.TEST_SERVICE) {
332                 shortRlmParametersBitString = Optional.of(bits.readBitString(15));
333                 // skip parity bit
334                 bits.skip(1);
335             } else { // OTHER
336                 shortRlmParametersBitString = Optional.of(bits.readBitString(15));
337             }
338             return Optional.of(new BeaconFeedback(rlmType1FeedbackReceived, rlmType2FeedbackReceived, rlsType,
339                     shortRlmParametersBitString));
340         } else {
341             return Optional.empty();
342         }
343     }

407     private static double readPositionCoordinate(Bits bits, int numDegreesBits) {
408         int sign = bits.readBoolean() ? -1 : 1;
409         int degrees = bits.readUnsignedInt(numDegreesBits);
410         // use BigDecimal so don't lose precision
411         BigDecimal d = readDecimalPart(bits);
412         d = d.add(BigDecimal.valueOf(degrees));
413         d = d.multiply(BigDecimal.valueOf(sign));
414         return d.doubleValue();
415     }

417     private static BigDecimal readDecimalPart(Bits bits) {
418         boolean[] decimalPart = bits.readBooleanArray(15);
419         BigDecimal d = BigDecimal.ZERO;
420         BigDecimal f = BigDecimal.ONE;
421         for (int i = 0; i < decimalPart.length; i++) {
422             f = f.divide(BigDecimal.valueOf(2));
423             if (decimalPart[i]) {
424                 d = d.add(f);
425             }
426         }
427         return d;
428     }
