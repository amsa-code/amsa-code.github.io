3   public final class KeyPrettify {
4   
5       private KeyPrettify() {
6           // prevent instantiation
7       }
8   
9       public static String prettify(String key) {
10          return replaceTokens(separateWords(key));
11      }
12  
13      private static String separateWords(String key) {
14          StringBuilder s = new StringBuilder();
15          s.append(Character.toUpperCase(key.charAt(0)));
16          boolean previousIsDigit = Character.isDigit(key.charAt(0));
17          for (int i = 1; i < key.length(); i++) {
18              char ch = key.charAt(i);
19              if (Character.isUpperCase(ch) || Character.isDigit(ch) && !previousIsDigit) {
20                  s.append(" ");
21              }
22              s.append(ch);
23              previousIsDigit = Character.isDigit(ch);
24          }
25          return s.toString();
26      }
27  
28      private static String replaceTokens(String words) {
29          return words //
30                  .replaceAll("\\bRls\\b", "RLS")//
31                  .replaceAll("\\bRlm\\b", "RLM") //
32                  .replaceAll("\\bGnss\\b", "GNSS") //
33                  .replaceAll("\\bHdop\\b", "HDOP") //
34                  .replaceAll("\\bVdop\\b", "VDOP") //
35                  .replaceAll("\\bElt\\b", "ELT") //
36                  .replaceAll("\\bDt\\b", "DT");
37      }
38  }
