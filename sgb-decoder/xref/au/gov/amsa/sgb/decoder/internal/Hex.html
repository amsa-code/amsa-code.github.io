3   import java.util.HashMap;
4   import java.util.Locale;
5   import java.util.Map;
6   import java.util.stream.Collectors;
7   
8   import com.github.davidmoten.guavamini.Preconditions;

10  public final class Hex {
11  
12      private Hex() {
13          // prevent instantiation
14      }
15  
16      private static Map<String, String> map = createMap();
17      private static Map<String, String> inverse = createInverse();
18  
19      private static Map<String, String> createMap() {
20          Map<String, String> m = new HashMap<>();
21          m.put("0", "0000");
22          m.put("1", "0001");
23          m.put("2", "0010");
24          m.put("3", "0011");
25          m.put("4", "0100");
26          m.put("5", "0101");
27          m.put("6", "0110");
28          m.put("7", "0111");
29          m.put("8", "1000");
30          m.put("9", "1001");
31          m.put("A", "1010");
32          m.put("B", "1011");
33          m.put("C", "1100");
34          m.put("D", "1101");
35          m.put("E", "1110");
36          m.put("F", "1111");
37          return m;
38      }
39  
40      private static Map<String, String> createInverse() {
41          return map.entrySet().stream().collect(Collectors.toMap(entry -> entry.getValue(), entry -> entry.getKey()));
42      }
43  
44      public static String hexToBinary(String s) {
45          char[] hex = s.toUpperCase(Locale.ENGLISH).toCharArray();
46          StringBuilder b = new StringBuilder();
47          for (char h : hex) {
48              b.append(map.get(String.valueOf(h)));
49          }
50          return b.toString();
51      }
52  
53      public static String bitStringToHex(String bitString) {
54          Preconditions.checkArgument(bitString.length() % 4 == 0);
55          StringBuilder b = new StringBuilder();
56          for (int i = 0; i < bitString.length(); i += 4) {
57              b.append(inverse.get(bitString.substring(i, i + 4)));
58          }
59          return b.toString();
60      }

62  }
