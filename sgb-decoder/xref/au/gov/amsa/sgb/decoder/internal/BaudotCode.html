62  
63      private static Integer val(String s) {
64          return Integer.parseInt(s, 2);
65      }
66  
67      public static char toChar(int code) {
68          Character ch = MAP.get(code);
69          if (ch == null) {
70              throw new IllegalArgumentException("character not found for decimal BaudotCode " + code);
71          } else {
72              return ch;
73          }
74      }
75  
76      public static char toCharFromShortCode(int shortCode) {
77          return toChar(shortCode + 32);
78      }
79  
80      @VisibleForTesting
81      static int size() {
82          return MAP.size();
83      }
