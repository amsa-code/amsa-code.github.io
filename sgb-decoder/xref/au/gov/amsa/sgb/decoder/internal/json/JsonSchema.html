25  /**
26   * Simplified JSON schema generator only targeting Detection class and
27   * its dependents. Honours Jackson {@link JsonIgnore} annotations on fields.
28   *
29   * <p>
30   * Limitations include:
31   * <ul>
32   * <li>arrays not supported (not required for Detection class)</li>
33   * <li>fields are considered {@code required} if not typed using
34   * {@link Optional}</li>
35   * </ul>
36   */
37  public final class JsonSchema {
38  
39      private static final char DQ = '"';
40      private static final String COLON = " : ";
41      private static final String COMMA = ",";
42  
43      private JsonSchema() {
44          // prevent instantiation
45      }
46  
47      /**
48       * Returns the JSON Schema for the heirarchy of classes pointed to by
49       * {@code classes}. Targets Detection class only (but might be extended
50       * in the future for more). Any subclasses within the heirarchy should be
51       * mentioned in {@code subclasses} so that the appropriate JSON Schema
52       * structures are produced. Subclasses should include a discriminator field that
53       * allows users to differentiate the JSON representations.
54       *
55       * @param cls        root class to be converted into a JSON Schema
56       * @param subclasses maps classes to a list of their subclasses
57       * @param schemaId   value to be used in the {@code $id} field.
58       * @return JSON Schema
59       */
60      public static String generateSchema(Class<?> cls, Map<Class<?>, List<Class<?>>> subclasses, String schemaId) {
61          // use all private fields to generate schema
62          Map<String, Definition> clsNameDefinitions = new HashMap<>();
63          collectDefinitions(cls, clsNameDefinitions, subclasses);
64          StringBuilder s = new StringBuilder();
65          addPreamble(schemaId, s);
66          addDefinitions(clsNameDefinitions, s);
67          s.append(COMMA + quoted("$ref") + COLON + quoted(toRef(cls)));
68          return "{" + s.toString() + "}";
69      }
70  
71      private static void addDefinitions(Map<String, Definition> clsNameDefinitions, StringBuilder s) {
72          s.append(quoted("definitions") + COLON + "{");
73          s.append(clsNameDefinitions.values() //
74                  .stream() //
75                  .map(x -> x.json) //
76                  .collect(Collectors.joining(",")));
77          s.append("}");
78      }
79  
80      private static void addPreamble(String schemaId, StringBuilder s) {
81          add(s, "$id", schemaId);
82          s.append(COMMA);
83          add(s, "$schema", "http://json-schema.org/draft/2019-09/schema");
84          s.append(COMMA);
85      }
86  
87      private static final class Definition {
88          private final String json;
89  
90          Definition(String json) {
91              this.json = json;
92          }
93      }

95      private static void collectDefinitions(Class<?> cls, Map<String, Definition> clsNameDefinitions,
96              Map<Class<?>, List<Class<?>>> subclasses) {
97          collectDefinitions(cls, clsNameDefinitions, new HashSet<String>(), subclasses);
98      }

100     private static void collectDefinitions(Class<?> cls, Map<String, Definition> clsNameDefinitions,
101             Set<String> classesAlreadyProcessed, Map<Class<?>, List<Class<?>>> subclasses) {
102         JsonType t = toJsonType(cls.getName());
103         if (t.typeName.equals("object") && !classesAlreadyProcessed.contains(cls.getName())) {
104             classesAlreadyProcessed.add(cls.getName());
105             // will be an implementation of HasFormatter
106             fields(cls).forEach(f -> {
107                 JsonType type = toJsonType(f.javaType());
108                 if (type.typeName.equals("object")) {
109                     collectDefinitions(toClass(f.javaType()), clsNameDefinitions, classesAlreadyProcessed, subclasses);
110                 } else if (type.typeName.equals("string") && !type.enumeration.isEmpty()) {
111                     // TODO use classesAlreadyProcessed
112                     StringBuilder json = new StringBuilder();
113                     json.append(quoted(definitionName(f.javaType())) + COLON + "{");
114                     add(json, "type", "string");
115                     json.append(", ");
116                     json.append(quoted("enum") + COLON);
117                     json.append(
118                             "[" + type.enumeration.stream().map(Util::quoted).collect(Collectors.joining(COMMA)) + "]");
119                     json.append("}");
120                     clsNameDefinitions.put(f.javaType(), new Definition(json.toString()));
121                 }
122             });
123             StringBuilder json = new StringBuilder();
124             json.append(quoted(definitionName(cls)) + COLON + "{");
125             List<Class<?>> list = subclasses.get(cls);
126             if (list != null) {
127                 StringBuilder s = new StringBuilder();
128                 // cls must be an interface because we don't use class inheritance
129                 for (Class<?> c : list) {
130                     collectDefinitions(c, clsNameDefinitions, subclasses);
131                     if (s.length() > 0) {
132                         s.append(", ");
133                     }
134                     s.append("{" + quoted("$ref") + COLON + quoted(toRef(c)) + "}");
135 
136                 }
137                 String types = "[" + s.toString() + "]";
138                 json.append(quoted("oneOf") + COLON + types);
139             } else {
140                 json.append(quoted("type") + COLON + quoted("object"));
141             }
142 
143             String properties = properties(cls);
144             if (!properties.isEmpty()) {
145                 json.append(", ");
146                 json.append(properties);
147             }
148 
149             String required = fields(cls) //
150                     .filter(f -> f.isRequired()) //
151                     .map(f -> quoted(f.name())) //
152                     .collect(Collectors.joining(", "));
153 
154             if (!required.isEmpty()) {
155                 json.append(COMMA);
156                 json.append(quoted("required") + COLON + "[" + required + "]");
157             }
158             json.append("}");
159             clsNameDefinitions.put(cls.getName(), new Definition(json.toString()));
160         }
161     }

163     private static Stream<MyField> fields(Class<?> cls) {
164         return Arrays.stream(cls.getDeclaredFields()) //
165                 .filter(f -> !isStatic(f)) //
166                 .filter(f -> !ignore(f)) //
167                 .map(JsonSchema::toMyField);
168     }
169 
170     private static boolean ignore(Field f) {
171         return f.getAnnotation(JsonIgnore.class) != null;
172     }

174     @VisibleForTesting
175     static Class<?> toClass(String javaClassName) {
176         try {
177             return Class.forName(javaClassName);
178         } catch (ClassNotFoundException e) {
179             throw new RuntimeException(e);
180         }
181     }

183     private static String generateDefinition(MyField f) {
184         StringBuilder b = new StringBuilder();
185         b.append(DQ);
186         b.append(f.name());
187         b.append(DQ);
188         b.append(" : ");
189         b.append("{");
190         JsonType t = toJsonType(f.javaType());
191         if (t.typeName.equals("object") || !t.enumeration.isEmpty()) {
192             add(b, "$ref", toRef(f.javaType()));
193         } else if (t.typeName.equals("time")) {
194             add(b, "type", "string");
195             b.append(COMMA);
196             add(b, "format", "time");
197         } else {
198             add(b, "type", t.typeName);
199         }
200 
201         b.append("}");
202         return b.toString();
203     }

205     private static String toRef(Class<?> cls) {
206         return "#/definitions/" + definitionName(cls);
207     }
208 
209     private static String toRef(String javaClassName) {
210         return "#/definitions/" + definitionName(javaClassName);
211     }

213     private static String definitionName(Class<?> cls) {
214         return definitionName(cls.getName());
215     }

217     private static String definitionName(String javaClassName) {
218         return simpleName(javaClassName);
219     }

221     @VisibleForTesting
222     static String simpleName(String javaClassName) {
223         int i = javaClassName.lastIndexOf('.');
224         if (i == -1) {
225             return javaClassName;
226         } else {
227             return javaClassName.substring(i + 1);
228         }
229     }

231     private static String properties(Class<?> cls) {
232         String content = fields(cls) //
233                 .map(JsonSchema::generateDefinition) //
234                 .collect(Collectors.joining(","));
235         if (content.trim().isEmpty()) {
236             return "";
237         } else {
238             return quoted("properties") + COLON + "{" + content //
239                     + "}";
240         }
241     }

243     private static boolean isStatic(Field f) {
244         return Modifier.isStatic(f.getModifiers());
245     }
246 
247     private static MyField toMyField(Field field) {
248         Class<?> t = field.getType();
249         final boolean required;
250         final String javaType;
251         if (t.equals(Optional.class)) {
252             ParameterizedType p = (ParameterizedType) field.getGenericType();
253             javaType = p.getActualTypeArguments()[0].getTypeName();
254             required = false;
255         } else {
256             javaType = t.getName();
257             required = true;
258         }
259         return new MyField(field.getName(), javaType, required);
260     }

262     private static void add(StringBuilder b, String key, String value) {
263         b.append(quoted(key) + COLON + quoted(value));
264     }

266     private static final Map<String, String> JAVA_TYPE_TO_JSON_TYPE = createJavaTypeToJsonTypeMap();

268     private static final class JsonType {
269         private final String typeName;
270         private final List<String> enumeration;
271 
272         private JsonType(String name, List<String> enumeration) {
273             this.typeName = name;
274             this.enumeration = enumeration;
275         }
276     }
277 
278     private static JsonType toJsonType(String javaType) {
279         String t = JAVA_TYPE_TO_JSON_TYPE.get(javaType);
280         if (t != null) {
281             return new JsonType(t, Collections.emptyList());
282         } else {
283             Class<?> cls = toClass(javaType);
284             if (cls.isEnum()) {
285                 List<String> list = Arrays.stream(cls.getEnumConstants()).map(x -> x.toString())
286                         .collect(Collectors.toList());
287                 return new JsonType("string", list);
288             } else {
289                 return new JsonType("object", Collections.emptyList());
290             }
291         }
292     }

294     private static Map<String, String> createJavaTypeToJsonTypeMap() {
295         Map<String, String> map = new HashMap<>();
296         map.put(Boolean.class.getName(), "boolean");
297         map.put("boolean", "boolean");
298         map.put(Integer.class.getName(), "integer");
299         map.put("int", "integer");
300         map.put(Double.class.getName(), "number");
301         map.put("double", "number");
302         map.put(String.class.getName(), "string");
303         map.put(OffsetTime.class.getName(), "time");
304         return map;
305     }
