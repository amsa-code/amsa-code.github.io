3   import java.util.Optional;
4   
5   import com.github.davidmoten.guavamini.Preconditions;
6   
7   public final class RadioCallSign implements VesselId {
8   
9       // used for serialization
10      private final VesselIdType vesselIdType = VesselIdType.RADIO_CALL_SIGN;
11      private final Optional<String> value;
12  
13      public RadioCallSign(String value) {
14          Preconditions.checkNotNull(value);
15          String s = value.trim();
16          if (s.isEmpty()) {
17              this.value = Optional.empty();
18          } else {
19              this.value = Optional.of(s);
20          }
21      }
22  
23      public VesselIdType vesselIdType() {
24          return vesselIdType;
25      }
26  
27      public Optional<String> value() {
28          return value;
29      }
30  
31  }
